17AELogEntry* AELogEntry::makeQueue(const std::size_t amt, AELogEntry* oldqueue) {
18 //damn, amt is really 0; get null!
19 if (amt == 0) {
20 throw std::runtime_error("queue size to allocate is 0!");

25 leptr[i].m_lepNextNode = leptr + i + 1; //set the next node pointers for our linked list

33 }

66 if (this->m_trdWriter.joinable()) {

91void AELogger::writeToLog(const std::string& logmessg, const cint logtype, const std::string& logmodule) {
92 if (!this->isOpen()) {
93 return; // file's closed/closing!
94 }
95
96 if (this->m_ullFilledCount > this->m_ullQueueSize) {
97 std::lock_guard<std::mutex> lock(this->m_mtxAllocLock);
98 ullint qsize = this->m_ullQueueSize / 2;
99
100 //make new queue
101 AELogEntry* newQueue = AELogEntry::makeQueue(qsize, this->m_lepQueue);
102 //add the new queue to the vector
103 this->m_ullQueueSize += qsize;
104 this->m_lepLastNode->m_lepNextNode = newQueue;
105 this->m_lepLastNode = newQueue + qsize - 1;
106 this->m_vAllocTable.push_back({ this->m_ullQueueSize, newQueue });
107 this->writeToLogDebug("The queue was too small, resized it to " + std::to_string(this->m_ullQueueSize) + " entries", AELOG_TYPE_WARN, this->m_sModulename);
108
109
110 }
111
113 if (logmessg.empty()) {
114 return; //na'ah, no empty messages
115 }
116
117 this->m_ullFilledCount.fetch_add(1);
118
119 //implementation: the allocation vector!
120 //instead of it having only the pointers to the allocated memory chunks of the log entries
121 //it would have the std::pair instead: pointer to chunk and the total amount of entries
122 //ex: {0xDEAD, 1024}, {0xBEEF, 1096} -> first pointer holds 1024 total entries, second holds 1096-1024=62 entries
123 //and instead of changing an atomic pointer, we'll increment a number, "current node number"
124 //and then check if the number(modulo of it with the m_ullQueueSize) is less than the 1st allocation in the vector, then 2nd, etc..untill we find a match
125 //then, to get the pointer to current node -> pointer of that allocation + number
126
127
128 //increment node number and get the pointer
129 AELogEntry* ptr = this->ptrFromIndex(m_ullNodeNumber++);
130 while (ptr->m_ullOrderNum != AELOG_ENTRY_INVALID_ORDERNUM && ptr->m_bStatus != AELOG_ENTRY_STATUS_INVALID) {
131 ptr = this->ptrFromIndex(m_ullNodeNumber++); //if current node is filled -> continue looking for unpopulated one
132 }
133
134
135 //populating it now!
136 ptr->m_ullOrderNum = this->m_ullLogOrderNum.fetch_add(1);
137 ptr->m_bStatus = AELOG_ENTRY_STATUS_SETTING; //alright boys, we're setting this one up
138
139 ptr->m_tmLogTime = std::time(NULL);
140 memcpy(ptr->m_sLogMessage, logmessg.c_str(), (logmessg.size() <= 511) ? logmessg.size() : 511);
141 memcpy(ptr->m_sModuleName, logmodule.c_str(), (logmodule.size() <= 32) ? logmodule.size() : 31);
142 ptr->m_cLogType = logtype;
143 ptr->m_bStatus = AELOG_ENTRY_STATUS_READY;
144}

195AELogEntry* AELogger::ptrFromIndex(ullint num) {
196 num %= m_ullQueueSize;
197 ullint prevSize = 0;
198 for (std::size_t i = 0; i < this->m_vAllocTable.size(); i++) {
199 if (this->m_vAllocTable[i].first > num) {
200 return this->m_vAllocTable[i].second + (num - prevSize);
