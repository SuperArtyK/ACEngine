52cint AELogger::startWriter(void) {
53 this->writeToLog("Attempting to start the log-writing thread", AELOG_TYPE_INFO, this->m_sModulename);
54 if (this->m_trdWriter.joinable()) {
55 this->writeToLog("The log-writing thread is already running!", AELOG_TYPE_WARN, this->m_sModulename);
56 return AELOG_ERR_THREAD_ALREADY_RUNNING; //we already are writing, dummy;
57 }
58
59 this->m_bRunTrd = true;
60 this->m_trdWriter = std::thread(&AELogger::logWriterThread, this);
61 if (!this->m_trdWriter.joinable()) {
62 this->writeToLog("Could not start AELogger thread!! (exception)", AELOG_TYPE_FATAL_ERROR, this->m_sModulename);
63 this->closeLog();
64 throw std::runtime_error("Could not start AELogger thread!");
65 return AELOG_ERR_UNABLE_START_THREAD;
66 }
67 return AELOG_ERR_NOERROR;
68}

83cint AELogger::writeToLog(const std::string_view logmessg, const cint logtype, const std::string_view logmodule) {
84 if (this->isClosed()) {
85 return AEFW_ERR_FILE_NOT_OPEN; // file's closed/closing!
86 }
87
88 // The only reason this function isn't shortcutted to a writeToLog with an entry
89 // is simply because it is doing the same thing anyway
90 // copying string data to a log entry
91 // and we'll need to do the same thing if we want to use the 2nd version of writeToLog
92
94
95 // check for invalid arguments
96 if (logmessg.empty() || logmodule.empty() || !ace::utils::isInRange<cint>(AELOG_TYPE_DEBUG, AELOG_TYPE_FATAL_ERROR, logtype) || //empty/invalid stuff
97 !ace::utils::isAlNumUs(logmodule)) { //log module stuff
98 return AELOG_ERR_INVALID_ENTRY_DATA;
99 }
100
101 this->m_ullFilledCount++;
102
103
104 if (this->m_ullFilledCount > this->m_ullQueueSize) {
105 const std::lock_guard<std::mutex> lock(this->m_mtxAllocLock);
106 const ullint qsize = this->m_ullQueueSize / 2;
107
108 //make new queue
109 AELogEntry* newQueue = AELogEntry::makeQueue(qsize, this->m_lepQueue);
110 //add the new queue to the vector
111 this->m_ullQueueSize += qsize;
112 this->m_lepLastNode->m_pNextNode = newQueue;
113 this->m_lepLastNode = newQueue + qsize - 1;
114 this->m_vAllocTable.emplace_back(this->m_ullQueueSize, newQueue);
115 this->writeToLogDebug("The queue was too small, resized it to " + std::to_string(this->m_ullQueueSize) + " entries", this->m_sModulename);
116 }
117
118
119 //implementation: the allocation vector!
120 //instead of it having only the pointers to the allocated memory chunks of the log entries
121 //it would have the std::pair instead: pointer to chunk and the total amount of entries
122 //ex: {0xDEAD, 1024}, {0xBEEF, 1096} -> first pointer holds 1024 total entries, second holds 1096-1024=62 entries
123 //and instead of changing an atomic pointer, we'll increment a number, "current node number"
124 //and then check if the number(modulo of it with the m_ullQueueSize) is less than the 1st allocation in the vector, then 2nd, etc..untill we find a match
125 //then, to get the pointer to current node -> pointer of that allocation + number
126
127
128 //increment node number and get the pointer
129 AELogEntry* ptr = this->ptrFromIndex(m_ullNodeNumber++);
130 while (ptr->m_cStatus != AELE_STATUS_INVALID) {
131 ptr = this->ptrFromIndex(m_ullNodeNumber++); //if current node is filled -> continue looking for unpopulated one
132 }
133
134
135 //populating it now!
136 ptr->m_cStatus = AELE_STATUS_SETTING; //alright boys, we're setting this one up
137
138 ptr->m_tmLogTime = std::time(nullptr);
139 memcpy(ptr->m_sLogMessage, logmessg.data(), (logmessg.size() > AELE_MESSAGE_SIZE) ? AELE_MESSAGE_SIZE : logmessg.size());
140 memcpy(ptr->m_sModuleName, logmodule.data(), (logmodule.size() > AELE_MODULENAME_SIZE) ? AELE_MODULENAME_SIZE : logmodule.size());
141 ptr->m_cLogType = logtype;
142 ptr->m_cStatus = AELE_STATUS_READY;
143 return AELOG_ERR_NOERROR;
144}

152

254AELogEntry* AELogger::ptrFromIndex(ullint num) noexcept {
255 num %= m_ullQueueSize;
256 ullint prevSize = 0;
257 for (std::size_t i = 0; i < this->m_vAllocTable.size(); i++) {
258 if (this->m_vAllocTable[i].first > num) {
259 return this->m_vAllocTable[i].second + (num - prevSize);
260 }
261 prevSize = this->m_vAllocTable[i].first;
262 }
263 return m_lepQueue;
264}
