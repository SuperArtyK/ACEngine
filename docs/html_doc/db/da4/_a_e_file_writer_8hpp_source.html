525inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) {
526 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
527
528 //check for the opened file here, before potentially setting 300+ digits for nothing
529 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
530
531 if constexpr (IS_SAME_NOC(T, float)) { // float
532 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
533 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
534 return this->writeString(buf, false, useAutoFlush);
535 }
536 else if constexpr (IS_SAME_NOC(T, double)) { // double
537 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
538 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
539 return this->writeString(buf, false, useAutoFlush);
540 }
541 else { // long double
542 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
543 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
544 return this->writeString(buf, false, useAutoFlush);
545 }

549template<typename T>
550inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
551
552 //open file?
553 if (!this->m_fpFilestr) {
554 this->m_cLastError = AEFW_ERR_FILE_NOT_OPEN;
555 return AEFW_ERR_FILE_NOT_OPEN;
556 }
557
558 // check for bool and char first
559 // because they trip the std::is_integral<T>
560 // and we dont want that
561
562 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
563 return this->writeBool(var);
564 }
565 // check also for unsigned char - since they call the same function anyway
566 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
567 return this->writeByte(var, useAutoFlush);
568 }
569
570 // now everything else
571 else if constexpr (std::is_integral<T>::value) { //integral type
572 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
573 return this->writeUInt(var, useAutoFlush);
574 }
575 else { // or signed?
576 return this->writeInt(var, useAutoFlush);
577 }
578 }
579 else if constexpr (std::is_floating_point<T>::value) { // float types
580 return this->writeFloat(var, useAutoFlush);
581 }
582 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
583 return this->writeString(var, false, useAutoFlush);
584 }
585 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
586 return this->writeString(var, useAutoFlush);
587 }
588
589 // alright. If it's not fundamental types
590 // then maybe it's a binary data to be written
591
592 //pointer to data
593 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
594 return this->writeBytes(var, datasz, useAutoFlush);
595 }
596 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
597 return this->writeBytes(var, useAutoFlush);
598 }
599 //or...direct object -> pass it as reference
600 else {
601 return this->writeData_ref(var, sizeof(T), useAutoFlush);
602 }
603
604}
