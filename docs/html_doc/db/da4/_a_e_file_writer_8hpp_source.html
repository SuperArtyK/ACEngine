530inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) {
531 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
532
533 //check for the opened file here, before potentially setting 300+ digits for nothing
534 _AEFW_EXIT_ON_CLOSED_FILE;
535
536 if constexpr (IS_SAME_NOC(T, float)) { // float
537 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
538 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
539 return this->writeString(buf, false, useAutoFlush);
540 }
541 else if constexpr (IS_SAME_NOC(T, double)) { // double
542 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
543 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
544 return this->writeString(buf, false, useAutoFlush);
545 }
546 else { // long double
547 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
548 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
549 return this->writeString(buf, false, useAutoFlush);
550 }

554template<typename T>
555inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
556
557 //open file?
558 if (!this->m_fpFilestr) {
559 this->m_cLastError = AEFW_ERR_FILE_NOT_OPEN;
560 return AEFW_ERR_FILE_NOT_OPEN;
561 }
562
563 // check for bool and char first
564 // because they trip the std::is_integral<T>
565 // and we dont want that
566
567 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
568 return this->writeBool(var);
569 }
570 // check also for unsigned char - since they call the same function anyway
571 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
572 return this->writeByte(var, useAutoFlush);
573 }
574
575 // now everything else
576 else if constexpr (std::is_integral<T>::value) { //integral type
577 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
578 return this->writeUInt(var, useAutoFlush);
579 }
580 else { // or signed?
581 return this->writeInt(var, useAutoFlush);
582 }
583 }
584 else if constexpr (std::is_floating_point<T>::value) { // float types
585 return this->writeFloat(var, useAutoFlush);
586 }
587 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
588 return this->writeString(var, false, useAutoFlush);
589 }
590 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
591 return this->writeString(var, useAutoFlush);
592 }
593
594 // alright. If it's not fundamental types
595 // then maybe it's a binary data to be written
596
597 //pointer to data
598 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
599 return this->writeBytes(var, datasz, useAutoFlush);
600 }
601 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
602 return this->writeBytes(var, useAutoFlush);
603 }
604 //or...direct object -> pass it as reference
605 else {
606 return this->writeData_ref(var, sizeof(T), useAutoFlush);
607 }
608
609}
