553inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) {
554 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
555
556 //check for the opened file here, before potentially setting 300+ digits for nothing
557 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
558
559 if constexpr (IS_SAME_NOC(T, float)) { // float
560 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
561 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
562 return this->writeString(buf, false, useAutoFlush);
563 }
564 else if constexpr (IS_SAME_NOC(T, double)) { // double
565 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
566 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
567 return this->writeString(buf, false, useAutoFlush);
568 }
569 else { // long double
570 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
571 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
572 return this->writeString(buf, false, useAutoFlush);
573 }

577template<typename T>
578inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
579
580 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
581
582 // check for bool and char first
583 // because they trip the std::is_integral<T>
584 // and we dont want that
585
586 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
587 return this->writeBool(var);
588 }
589 // check also for unsigned char - since they call the same function anyway
590 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
591 return this->writeByte(var, useAutoFlush);
592 }
593
594 // now everything else
595 else if constexpr (std::is_integral<T>::value) { //integral type
596 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
597 return this->writeUInt(var, useAutoFlush);
598 }
599 else { // or signed?
600 return this->writeInt(var, useAutoFlush);
601 }
602 }
603 else if constexpr (std::is_floating_point<T>::value) { // float types
604 return this->writeFloat(var, useAutoFlush);
605 }
606 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
607 return this->writeString(var, false, useAutoFlush);
608 }
609 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
610 return this->writeString(var, useAutoFlush);
611 }
612
613 // alright. If it's not fundamental types
614 // then maybe it's a binary data to be written
615
616 //pointer to data
617 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
618 return this->writeBytes(var, datasz, useAutoFlush);
619 }
620 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
621 return this->writeBytes(var, useAutoFlush);
622 }
623 //or...direct object -> pass it as reference
624 else {
625 return this->writeData_ref(var, sizeof(T), useAutoFlush);
626 }
627
628}
