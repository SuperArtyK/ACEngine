597template<typename T>
598inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) {
599 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
600
601 //check for the opened file here, before potentially setting 300+ digits for nothing
602 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
603
604 if constexpr (IS_SAME_NOC(T, float)) { // float
605 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
606 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
607 return this->writeString(buf, false, useAutoFlush);
608 }
609 else if constexpr (IS_SAME_NOC(T, double)) { // double
610 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
611 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
612 return this->writeString(buf, false, useAutoFlush);
613 }
614 else { // long double
615 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
616 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
617 return this->writeString(buf, false, useAutoFlush);
618 }
619}

621// write stuff dependant on data
622template<typename T>
623inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
624
625 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
626
627 // check for bool and char first
628 // because they trip the std::is_integral<T>
629 // and we dont want that
630
631 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
632 return this->writeBool(var);
633 }
634 // check also for unsigned char - since they call the same function anyway
635 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
636 return this->writeByte(var, useAutoFlush);
637 }
638
639 // now everything else
640 else if constexpr (std::is_integral<T>::value) { //integral type
641 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
642 return this->writeUInt(var, useAutoFlush);
643 }
644 else { // or signed?
645 return this->writeInt(var, useAutoFlush);
646 }
647 }
648 else if constexpr (std::is_floating_point<T>::value) { // float types
649 return this->writeFloat(var, useAutoFlush);
650 }
651 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
652 return this->writeString(var, false, useAutoFlush);
653 }
654 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
655 return this->writeString(var, useAutoFlush);
656 }
657
658 // alright. If it's not fundamental types
659 // then maybe it's a binary data to be written
660
661 //pointer to data
662 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
663 return this->writeBytes(var, datasz, useAutoFlush);
664 }
665 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
666 return this->writeBytes(var, useAutoFlush);
667 }
668 //or...direct object -> pass it as reference
669 else {
670 return this->writeData_ref(var, sizeof(T), useAutoFlush);
671 }
672
673}
