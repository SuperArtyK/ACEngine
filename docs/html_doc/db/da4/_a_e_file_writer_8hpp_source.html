591// write stuff dependant on data
592template<typename T>
593inline void AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
594
595 //open file?
596 if (!this->m_fpFilestr) {
597 this->m_cLastError = AEFW_ERR_NOT_OPEN;
598 return;
599 }
600
601 // check for bool and char first
602 // because they trip the std::is_integral<T>
603 // and we dont want that
604
605 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
606 this->writeBool(var);
607 }
608 // check also for unsigned char - since they call the same function anyway
609 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
610 this->writeByte(var, useAutoFlush);
611 }
612
613 // now everything else
614 else if constexpr (std::is_integral<T>::value) { //integral type
615 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
616 this->writeUInt(var, useAutoFlush);
617 }
618 else { // or signed?
619 this->writeInt(var, useAutoFlush);
620 }
621 }
622 else if constexpr (std::is_floating_point<T>::value) { // float types
623 this->writeFloat(var, useAutoFlush);
624 }
625 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
626 this->writeString(var, false, useAutoFlush);
627 }
628 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
629 this->writeString(var, useAutoFlush);
630 }
631
632 // alright. If it's not fundamental types
633 // then maybe it's a binary data to be written
634
635 //pointer to data
636 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
637 this->writeBytes(var, datasz, useAutoFlush);
638 }
639 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
640 this->writeBytes(var, useAutoFlush);
641 }
642 //or...direct object -> pass it as reference
643 else {
644 this->writeData_ref(var, sizeof(T), useAutoFlush);
645 }

649// open file with flags
650// uses const char* as all normal string function do
651bool AEFileWriter::open(const char* str, const ucint flags, const ullint af_interval) {
652
653 if (this->m_fpFilestr) { // open already -> pls close
654 this->closeFile();
655 }
656
657 if (!strlen(str)) {
658 this->m_sFilename.clear();
659 this->m_cLastError = AEFW_ERR_FILE_NAME_EMPTY;
660 return false;
661 }
662
663
664 this->m_sFilename = str;
665 std::size_t found = this->m_sFilename.rfind("/");
666 if (found != std::string::npos) {
667 std::filesystem::create_directories(this->m_sFilename.substr(0, found));
668 }
669
670 switch (flags) {
671
672 case AEFW_FLAG_APPEND://cursor at end, allow changing cursor position
673 this->openDirectly(str, "r+b");
674 if (!this->m_fpFilestr) {//couldnt open, force create
675 this->openDirectly(str, "wb");
676 if (!this->m_fpFilestr) {
677 break; // some other error
678 }
679 }
680 fseek(this->m_fpFilestr, 0L, SEEK_END);
681 break;
682
683 case AEFW_FLAG_APPEND_NO_CURSOR_MOVE://only appending, not allowing to change cursor position
684 this->openDirectly(str, "ab");
685 break;
686
687
688 case AEFW_FLAG_NOFLAGS://no flags, defaulting to truncation
689 case AEFW_FLAG_TRUNCATE://truncate file
690 this->openDirectly(str, "wb");
691 break;
692
693 default:
694 this->m_cLastError = AEFW_ERR_FILE_WRONG_FLAG; //wrong flag, not opening the file
695 return false;
696 }
697
698 // last check, to see if everything is okay
699 if (!this->m_fpFilestr) {
700 this->m_cLastError = AEFW_ERR_FILE_OPEN_ELSE; //file is still somehow nonexistent
701 return false;
702 }
703
704 //then everything is good,
705 this->m_ucFlags = flags;
706 this->m_ullFlushInterval = af_interval;
707 return true;
708}
