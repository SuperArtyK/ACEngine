568inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) {
569 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
570
571 //check for the opened file here, before potentially setting 300+ digits for nothing
572 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
573
574 if constexpr (IS_SAME_NOC(T, float)) { // float
575 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
576 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
577 return this->writeString(buf, false, useAutoFlush);
578 }
579 else if constexpr (IS_SAME_NOC(T, double)) { // double
580 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
581 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
582 return this->writeString(buf, false, useAutoFlush);
583 }
584 else { // long double
585 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
586 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
587 return this->writeString(buf, false, useAutoFlush);
588 }
589}

591// write stuff dependant on data
592template<typename T>
593inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
594
595 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
596
597 // check for bool and char first
598 // because they trip the std::is_integral<T>
599 // and we dont want that
600
601 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
602 return this->writeBool(var);
603 }
604 // check also for unsigned char - since they call the same function anyway
605 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
606 return this->writeByte(var, useAutoFlush);
607 }
608
609 // now everything else
610 else if constexpr (std::is_integral<T>::value) { //integral type
611 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
612 return this->writeUInt(var, useAutoFlush);
613 }
614 else { // or signed?
615 return this->writeInt(var, useAutoFlush);
616 }
617 }
618 else if constexpr (std::is_floating_point<T>::value) { // float types
619 return this->writeFloat(var, useAutoFlush);
620 }
621 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
622 return this->writeString(var, false, useAutoFlush);
623 }
624 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
625 return this->writeString(var, useAutoFlush);
626 }
627
628 // alright. If it's not fundamental types
629 // then maybe it's a binary data to be written
630
631 //pointer to data
632 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
633 return this->writeBytes(var, datasz, useAutoFlush);
634 }
635 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
636 return this->writeBytes(var, useAutoFlush);
637 }
638 //or...direct object -> pass it as reference
639 else {
640 return this->writeData_ref(var, sizeof(T), useAutoFlush);
641 }
642
643}
