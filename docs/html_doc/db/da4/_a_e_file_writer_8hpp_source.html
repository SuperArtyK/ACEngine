584inline cint AEFileWriter::writeData_ptr(const void* cdata, const std::size_t dcount, const std::size_t dsize, const bool useAutoFlush) {
585
586 _AEFW_EXIT_ON_CLOSED_FILE;
587
588 if (!dcount || !dsize || !cdata) {
589 return AEFW_ERR_WRITE_ZERO_SIZE;
590 }
591
592 const std::size_t writestatus = fwrite(cdata, dsize, dcount, m_fpFilestr);
593 this->m_ullTotalWrites++;
594 if (useAutoFlush) {
595 this->autoFlush();
596 }
597
598 if (writestatus != dcount) {
599 return AEFW_ERR_WRITE_ERROR;
600 }
601
602 return AEFW_ERR_WRITE_SUCCESS;
603}

606template<typename T>
607inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
608
609 //open file?
610 if (!this->m_fpFilestr) {
611 this->m_cLastError = AEFW_ERR_FILE_NOT_OPEN;
612 return AEFW_ERR_FILE_NOT_OPEN;
613 }
614
615 // check for bool and char first
616 // because they trip the std::is_integral<T>
617 // and we dont want that
618
619 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
620 return this->writeBool(var);
621 }
622 // check also for unsigned char - since they call the same function anyway
623 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
624 return this->writeByte(var, useAutoFlush);
625 }
626
627 // now everything else
628 else if constexpr (std::is_integral<T>::value) { //integral type
629 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
630 return this->writeUInt(var, useAutoFlush);
631 }
632 else { // or signed?
633 return this->writeInt(var, useAutoFlush);
634 }
635 }
636 else if constexpr (std::is_floating_point<T>::value) { // float types
637 return this->writeFloat(var, useAutoFlush);
638 }
639 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
640 return this->writeString(var, false, useAutoFlush);
641 }
642 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
643 return this->writeString(var, useAutoFlush);
644 }
645
646 // alright. If it's not fundamental types
647 // then maybe it's a binary data to be written
648
649 //pointer to data
650 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
651 return this->writeBytes(var, datasz, useAutoFlush);
652 }
653 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
654 return this->writeBytes(var, useAutoFlush);
655 }
656 //or...direct object -> pass it as reference
657 else {
658 return this->writeData_ref(var, sizeof(T), useAutoFlush);
659 }
660
661}

663// open file with flags
664// uses const char* as all normal string function do
665int AEFileWriter::openFile(const char* str, const cint flags, const ullint af_interval) {
666
667 if (this->m_fpFilestr) { // open already -> pls close
668 this->closeFile();
669 }
670
671 if (!str || !std::strlen(str)) {
672 this->m_sFilename.clear();
673 this->m_cLastError = AEFW_ERR_FILE_NAME_EMPTY;
674 return AEFW_ERR_FILE_NAME_EMPTY;
675 }
676
677
678 this->m_sFilename = str;
679 std::size_t found = this->m_sFilename.rfind("/");
680 if (found != std::string::npos) {
681 std::filesystem::create_directories(this->m_sFilename.substr(0, found));
682 }
683
684 switch (flags) {
685
686 case AEFW_FLAG_APPEND://cursor at end, allow changing cursor position
687 this->m_fpFilestr = ace::utils::fopenCC(str, "r+b");
688 if (!this->m_fpFilestr) {//couldnt open, force create
689 this->m_fpFilestr = ace::utils::fopenCC(str, "wb");
690 if (!this->m_fpFilestr) {
691 break; // some other error
692 }
693 }
694 fseek(this->m_fpFilestr, 0L, SEEK_END);
695 break;
696
697 case AEFW_FLAG_APPEND_NO_CURSOR_MOVE://only appending, not allowing to change cursor position
698 this->m_fpFilestr = ace::utils::fopenCC(str, "ab");
699 break;
700
701
702 case AEFW_FLAG_NOFLAGS://no flags, defaulting to truncation
703 case AEFW_FLAG_TRUNCATE://truncate file
704 this->m_fpFilestr = ace::utils::fopenCC(str, "wb");
705 break;
706
707 default:
708 this->m_cLastError = AEFW_ERR_FILE_WRONG_FLAG; //wrong flag, not opening the file
709 return AEFW_ERR_FILE_WRONG_FLAG;
710 }
711
712 // last check, to see if everything is okay
713 if (!this->m_fpFilestr) {
714 this->m_cLastError = AEFW_ERR_FILE_OPEN_ELSE; //file is still somehow nonexistent
715 return AEFW_ERR_FILE_OPEN_ELSE;
716 }
717
718 //then everything is good,
719 this->m_ucFlags = flags;
720 this->m_ullFlushInterval = af_interval;
721 return AEFW_ERR_NOERROR;
722}
