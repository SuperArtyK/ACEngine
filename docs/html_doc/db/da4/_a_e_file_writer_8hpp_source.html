595// write stuff dependant on data
596template<typename T>
597inline void AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
598
599 //open file?
600 if (!this->m_fpFilestr) {
601 this->m_cLastError = AEFW_ERR_NOT_OPEN;
602 return;
603 }
604
605 // check for bool and char first
606 // because they trip the std::is_integral<T>
607 // and we dont want that
608
609 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
610 this->writeBool(var);
611 }
612 // check also for unsigned char - since they call the same function anyway
613 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
614 this->writeByte(var, useAutoFlush);
615 }
616
617 // now everything else
618 else if constexpr (std::is_integral<T>::value) { //integral type
619 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
620 this->writeUInt(var, useAutoFlush);
621 }
622 else { // or signed?
623 this->writeInt(var, useAutoFlush);
624 }
625 }
626 else if constexpr (std::is_floating_point<T>::value) { // float types
627 this->writeFloat(var, useAutoFlush);
628 }
629 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
630 this->writeString(var, false, useAutoFlush);
631 }
632 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
633 this->writeString(var, useAutoFlush);
634 }
635
636 // alright. If it's not fundamental types
637 // then maybe it's a binary data to be written
638
639 //pointer to data
640 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
641 this->writeBytes(var, datasz, useAutoFlush);
642 }
643 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
644 this->writeBytes(var, useAutoFlush);
645 }
646 //or...direct object -> pass it as reference
647 else {
648 this->writeData_ref(var, sizeof(T), useAutoFlush);
649 }
650
651}

653// open file with flags
654// uses const char* as all normal string function do
655bool AEFileWriter::open(const char* str, const ucint flags, const ullint af_interval) {
656
657 if (this->m_fpFilestr) { // open already -> pls close
658 this->closeFile();
659 }
660
661 if (!std::strlen(str)) {
662 this->m_sFilename.clear();
663 this->m_cLastError = AEFW_ERR_FILE_NAME_EMPTY;
664 return false;
665 }
666
667
668 this->m_sFilename = str;
669 std::size_t found = this->m_sFilename.rfind("/");
670 if (found != std::string::npos) {
671 std::filesystem::create_directories(this->m_sFilename.substr(0, found));
672 }
673
674 switch (flags) {
675
676 case AEFW_FLAG_APPEND://cursor at end, allow changing cursor position
677 this->openDirectly(str, "r+b");
678 if (!this->m_fpFilestr) {//couldnt open, force create
679 this->openDirectly(str, "wb");
680 if (!this->m_fpFilestr) {
681 break; // some other error
682 }
683 }
684 fseek(this->m_fpFilestr, 0L, SEEK_END);
685 break;
686
687 case AEFW_FLAG_APPEND_NO_CURSOR_MOVE://only appending, not allowing to change cursor position
688 this->openDirectly(str, "ab");
689 break;
690
691
692 case AEFW_FLAG_NOFLAGS://no flags, defaulting to truncation
693 case AEFW_FLAG_TRUNCATE://truncate file
694 this->openDirectly(str, "wb");
695 break;
696
697 default:
698 this->m_cLastError = AEFW_ERR_FILE_WRONG_FLAG; //wrong flag, not opening the file
699 return false;
700 }
701
702 // last check, to see if everything is okay
703 if (!this->m_fpFilestr) {
704 this->m_cLastError = AEFW_ERR_FILE_OPEN_ELSE; //file is still somehow nonexistent
705 return false;
706 }
707
708 //then everything is good,
709 this->m_ucFlags = flags;
710 this->m_ullFlushInterval = af_interval;
711 return true;
712}
