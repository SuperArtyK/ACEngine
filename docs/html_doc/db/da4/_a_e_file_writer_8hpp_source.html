594inline cint AEFileWriter::writeData_ptr(const void* cdata, const std::size_t dcount, const std::size_t dsize, const bool useAutoFlush) {
595
596 _AEFW_EXIT_ON_CLOSED_FILE;
597
598 if (!dcount || !dsize || !cdata) {
599 return AEFW_ERR_WRITE_ZERO_SIZE;
600 }
601
602 const std::size_t writestatus = fwrite(cdata, dsize, dcount, m_fpFilestr);
603 this->m_ullTotalWrites++;
604 if (useAutoFlush) {
605 this->autoFlush();
606 }
607
608 if (writestatus != dcount) {
609 return AEFW_ERR_WRITE_ERROR;
610 }
611
612 return AEFW_ERR_WRITE_SUCCESS;
613}

616template<typename T>
617inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
618
619 //open file?
620 if (!this->m_fpFilestr) {
621 this->m_cLastError = AEFW_ERR_FILE_NOT_OPEN;
622 return AEFW_ERR_FILE_NOT_OPEN;
623 }
624
625 // check for bool and char first
626 // because they trip the std::is_integral<T>
627 // and we dont want that
628
629 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
630 return this->writeBool(var);
631 }
632 // check also for unsigned char - since they call the same function anyway
633 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
634 return this->writeByte(var, useAutoFlush);
635 }
636
637 // now everything else
638 else if constexpr (std::is_integral<T>::value) { //integral type
639 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
640 return this->writeUInt(var, useAutoFlush);
641 }
642 else { // or signed?
643 return this->writeInt(var, useAutoFlush);
644 }
645 }
646 else if constexpr (std::is_floating_point<T>::value) { // float types
647 return this->writeFloat(var, useAutoFlush);
648 }
649 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
650 return this->writeString(var, false, useAutoFlush);
651 }
652 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
653 return this->writeString(var, useAutoFlush);
654 }
655
656 // alright. If it's not fundamental types
657 // then maybe it's a binary data to be written
658
659 //pointer to data
660 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
661 return this->writeBytes(var, datasz, useAutoFlush);
662 }
663 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
664 return this->writeBytes(var, useAutoFlush);
665 }
666 //or...direct object -> pass it as reference
667 else {
668 return this->writeData_ref(var, sizeof(T), useAutoFlush);
669 }

673// open file with flags
674// uses const char* as all normal string function do
675int AEFileWriter::openFile(const char* str, const cint flags, const ullint af_interval) {
676
677 if (this->m_fpFilestr) { // open already -> pls close
678 this->closeFile();
679 }
680
681 if (!str || !std::strlen(str)) {
682 this->m_sFilename.clear();
683 this->m_cLastError = AEFW_ERR_FILE_NAME_EMPTY;
684 return AEFW_ERR_FILE_NAME_EMPTY;
685 }
686
687
688 this->m_sFilename = str;
689 std::size_t found = this->m_sFilename.rfind("/");
690 if (found != std::string::npos) {
691 std::filesystem::create_directories(this->m_sFilename.substr(0, found));
692 }
693
694 switch (flags) {
695
696 case AEFW_FLAG_APPEND://cursor at end, allow changing cursor position
697 this->m_fpFilestr = ace::utils::fopenCC(str, "r+b");
698 if (!this->m_fpFilestr) {//couldnt open, force create
699 this->m_fpFilestr = ace::utils::fopenCC(str, "wb");
700 if (!this->m_fpFilestr) {
701 break; // some other error
702 }
703 }
704 fseek(this->m_fpFilestr, 0L, SEEK_END);
705 break;
706
707 case AEFW_FLAG_APPEND_NO_CURSOR_MOVE://only appending, not allowing to change cursor position
708 this->m_fpFilestr = ace::utils::fopenCC(str, "ab");
709 break;
710
711
712 case AEFW_FLAG_NOFLAGS://no flags, defaulting to truncation
713 case AEFW_FLAG_TRUNCATE://truncate file
714 this->m_fpFilestr = ace::utils::fopenCC(str, "wb");
715 break;
716
717 default:
718 this->m_cLastError = AEFW_ERR_FILE_WRONG_FLAG; //wrong flag, not opening the file
719 return AEFW_ERR_FILE_WRONG_FLAG;
720 }
721
722 // last check, to see if everything is okay
723 if (!this->m_fpFilestr) {
724 this->m_cLastError = AEFW_ERR_FILE_OPEN_ELSE; //file is still somehow nonexistent
725 return AEFW_ERR_FILE_OPEN_ELSE;
726 }
727
728 //then everything is good,
729 this->m_ucFlags = flags;
730 this->m_ullFlushInterval = af_interval;
731 return AEFW_ERR_NOERROR;
732}
