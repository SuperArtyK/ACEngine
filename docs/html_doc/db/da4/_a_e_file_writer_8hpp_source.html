425 /// <summary>
426 /// Returns size of the file in bytes.
427 /// @warning Fails and returns AEFW_ERR_OPEN_FILE_WRONG_FLAG, if the flag that was used to open the current file is AEFW_FLAG_APPEND_NO_CURSOR_MOVE
428 /// </summary>
429 /// <returns>File size in bytes if file is open; otherwise AEFW_ERR_FILE_NOT_OPEN</returns>
430 inline llint getFileSize(void) const noexcept {
431 _AEFW_EXIT_ON_CLOSED_FILE;
432 _AEFW_EXIT_ON_NO_CURSOR_MOVE;
433
434 const llint curpos = ftell(this->m_fpFilestr); // get current position
435 fseek(this->m_fpFilestr, 0, SEEK_END); // change pos to end of file
436 const llint endpos = ftell(this->m_fpFilestr); // get pos of file at the end; file size essentially
437 fseek(this->m_fpFilestr, curpos, SEEK_SET); // return to original pos
438 return endpos;
439 }

514 /// <summary>
515 /// Sets read cursor position to pos from origin.
516 /// @note If cursor is beyond EOF, it fills space between EOF and cursor with null-bytes when data is written.
517 /// @warning Fails and returns AEFW_ERR_OPEN_FILE_WRONG_FLAG, if the flag that was used to open the current file is AEFW_FLAG_APPEND_NO_CURSOR_MOVE
518 /// @note If origin is not SEEK_SET, SEEK_CUR or SEEK_END returns AEFR_ERR_READING_EOF
519 /// </summary>
520 /// <param name="pos">Position to be set to relative to origin (same as "offset" in fseek)</param>
521 /// <param name="origin">Relative origin for the operation. Google SEEK_SET, SEEK_CUR and SEEK_END for more details</param>

523 inline int setCursorPos(const llint pos, const int origin = SEEK_CUR) const noexcept {
524 _AEFW_EXIT_ON_CLOSED_FILE;
525 _AEFW_EXIT_ON_NO_CURSOR_MOVE;
526
527 if (origin != SEEK_SET && origin != SEEK_CUR && origin != SEEK_END) {
528 return AEFW_ERR_WRITE_ERROR;
529 }
530

532 }

597template<typename T>
598inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) noexcept requires(std::is_floating_point<T>::value == true) {
599 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
600
601 //check for the opened file here, before potentially setting 300+ digits for nothing
602 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
603
604 if constexpr (IS_SAME_NOCV(T, float)) { // float
605 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
606 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
607 return this->writeString(buf, false, useAutoFlush);
608 }
609 else if constexpr (IS_SAME_NOCV(T, double)) { // double
610 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
611 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
612 return this->writeString(buf, false, useAutoFlush);
613 }
614 else { // long double
615 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
616 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
617 return this->writeString(buf, false, useAutoFlush);
618 }

621// write stuff dependant on data
622template<typename T>
623inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) noexcept{
624
625 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
626
627 // check for bool and char first
628 // because they trip the std::is_integral<T>
629 // and we dont want that
630
631 if constexpr (IS_SAME_NOCV(T, bool)) { //bool type
632 return this->writeBool(var);
633 }
634 // check also for unsigned char - since they call the same function anyway
635 else if constexpr (IS_SAME_NOCV(T, char) || IS_SAME_NOCV(T, unsigned char)) { //char type
636 return this->writeByte(var, useAutoFlush);
637 }
638
639 // now everything else
640 else if constexpr (std::is_integral<T>::value) { //integral type
641 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
642 return this->writeUInt(var, useAutoFlush);
643 }
644 else { // or signed?
645 return this->writeInt(var, useAutoFlush);
646 }
647 }
648 else if constexpr (std::is_floating_point<T>::value) { // float types
649 return this->writeFloat(var, useAutoFlush);
650 }
651 else if constexpr (IS_SAME_NOCV(T, char*) || IS_SAME_NOCV(T, std::string)) { // strings
652 return this->writeString(var, false, useAutoFlush);
653 }
654 else if constexpr (IS_SAME_NOCV(T, std::vector<char>)) { // strings in form of a std::vector<char>
655 return this->writeString(var, useAutoFlush);
656 }
657
658 // alright. If it's not fundamental types
659 // then maybe it's a binary data to be written
660
661 //pointer to data
662 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
663 return this->writeBytes(var, datasz, useAutoFlush);
664 }
665 else if constexpr (IS_SAME_NOCV(T, std::vector<unsigned char>)) {
666 return this->writeBytes(var, useAutoFlush);
667 }
668 //or...direct object -> pass it as reference
669 else {
670 return this->writeData_ref(var, sizeof(T), useAutoFlush);
671 }
672
673}
