595inline cint AEFileWriter::writeFloat(const T num, const bool useAutoFlush) {
596 static_assert(std::is_floating_point<T>::value, "Cannot use non-float types in AEFileWriter::writeFloat!");
597
598 //check for the opened file here, before potentially setting 300+ digits for nothing
599 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
600
601 if constexpr (IS_SAME_NOC(T, float)) { // float
602 char buf[FLT_MAX_10_EXP + FLT_DIG + 4]{}; // juuust 1 more than max of float value, so with -FLT_MAX it will have a trailing null
603 snprintf(buf, sizeof(buf), "%.*f", FLT_DIG, num);
604 return this->writeString(buf, false, useAutoFlush);
605 }
606 else if constexpr (IS_SAME_NOC(T, double)) { // double
607 char buf[DBL_MAX_10_EXP + DBL_DIG + 4]{};
608 snprintf(buf, sizeof(buf), "%.*lf", DBL_DIG, num);
609 return this->writeString(buf, false, useAutoFlush);
610 }
611 else { // long double
612 char buf[LDBL_MAX_10_EXP + LDBL_DIG + 4]{};
613 snprintf(buf, sizeof(buf), "%.*Lf", LDBL_DIG, num);
614 return this->writeString(buf, false, useAutoFlush);
615 }

619template<typename T>
620inline cint AEFileWriter::write(const T& var, const size_t datasz, const bool useAutoFlush) {
621
622 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
623
624 // check for bool and char first
625 // because they trip the std::is_integral<T>
626 // and we dont want that
627
628 if constexpr (IS_SAME_NOC(T, bool)) { //bool type
629 return this->writeBool(var);
630 }
631 // check also for unsigned char - since they call the same function anyway
632 else if constexpr (IS_SAME_NOC(T, char) || IS_SAME_NOC(T, unsigned char)) { //char type
633 return this->writeByte(var, useAutoFlush);
634 }
635
636 // now everything else
637 else if constexpr (std::is_integral<T>::value) { //integral type
638 if constexpr (std::is_unsigned<T>::value) { // what about unsigned?
639 return this->writeUInt(var, useAutoFlush);
640 }
641 else { // or signed?
642 return this->writeInt(var, useAutoFlush);
643 }
644 }
645 else if constexpr (std::is_floating_point<T>::value) { // float types
646 return this->writeFloat(var, useAutoFlush);
647 }
648 else if constexpr (IS_SAME_NOC(T, char*) || IS_SAME_NOC(T, std::string)) { // strings
649 return this->writeString(var, false, useAutoFlush);
650 }
651 else if constexpr (IS_SAME_NOC(T, std::vector<char>)) { // strings in form of a std::vector<char>
652 return this->writeString(var, useAutoFlush);
653 }
654
655 // alright. If it's not fundamental types
656 // then maybe it's a binary data to be written
657
658 //pointer to data
659 else if constexpr (std::is_pointer<T>::value && datasz > 0) { //and check if data stream size is not 0
660 return this->writeBytes(var, datasz, useAutoFlush);
661 }
662 else if constexpr (IS_SAME_NOC(T, std::vector<unsigned char>)) {
663 return this->writeBytes(var, useAutoFlush);
664 }
665 //or...direct object -> pass it as reference
666 else {
667 return this->writeData_ref(var, sizeof(T), useAutoFlush);
668 }
669
670}
