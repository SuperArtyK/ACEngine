19/// <summary>
20/// @brief ArtyK's Engine Timer -- the engine's internal timer for, um, timing events or anything, with variable tick speed.
21/// It makes a separate thread that increments the tick number every set time period. For the delay, it uses AEFrame.
22///
23/// Just create it and look at the ticks of the timer from anywhere in the program...that you can access it.
24/// Hungarian notation is tm. (m_tmMyTimer)
25/// </summary>
26class AETimer : public __AEModuleBase<AETimer> {
27
28public:
29
30//constructors
31 /// <summary>
32 /// Class constructor -- starts the timer automatically
33 /// </summary>
34 /// <param name="tps">The amount of ticks per second, as the float</param>
35 explicit AETimer(const double tps = ENGINE_FPS) :
36 m_rDelay(tps), m_ullTicks(0), m_bRunTrd(true) {
37 this->startThread();
38 }
39
40 /// <summary>
41 /// Class constructor -- just assigns default delay (ENGINE_FPS) and doesn't start the thread
42 /// </summary>
43 AETimer(void) noexcept :
44 m_rDelay(ENGINE_FPS), m_ullTicks(0), m_bRunTrd(false)
45 {}
46
47 /// <summary>
48 /// Copy constructor -- Just copies the data and then starts the timer if original timer was started.
49 /// </summary>
50 /// <param name="tm">The original AETimer instance to make a copy of</param>
51 AETimer(const AETimer& tm) :
52 //COPYCONSTRUCT_MODULE(AETimer, tm),
53 m_rDelay(tm.m_rDelay), m_ullTicks(tm.m_ullTicks.load()), m_bRunTrd(tm.m_bRunTrd.load()) {
54
55 if (this->m_bRunTrd) {
56 this->startThread();
57 }
58 }
59
60 /// <summary>
61 /// The copy assignment operator. Just copies the data and then starts the timer if original timer was started.
62 /// </summary>
63 /// <param name="tm">The original AETimer instance to make a copy of</param>
64 /// <returns>Reference to the resulting AETimer copy</returns>
65 AETimer& operator=(const AETimer& tm) {
66 this->m_rDelay = tm.m_rDelay;
67 this->m_ullTicks.store(tm.m_ullTicks.load());
68 this->m_bRunTrd.store(tm.m_bRunTrd.load());
69 if (this->m_bRunTrd) {
70 this->startThread();
71 }
72 return *this;
73 }
74
75 /// <summary>
76 /// Class destructor
77 /// </summary>
78 ~AETimer(void) {
79 this->stopThread();
80 }

82 /// <summary>
83 /// Starts the timer thread and starts tick counting
84 /// @note Does nothing if thread is already started
85 /// </summary>
86 inline void startThread(void) {
87 if (this->m_trdCounting.joinable()) {
88 return; // we're already running, silly
89 }
90
91 this->m_bRunTrd = true;
92 this->m_trdCounting = std::thread(&AETimer::tickTimer, this);
93 if (!this->m_trdCounting.joinable()) {
94 throw std::runtime_error("Could not start AETimer thread!");
95 }
96 }
97
98 /// <summary>
99 /// Stops the timer thread and stops tick counting
100 /// </summary>
101 inline void stopThread(void) {
102 if (!this->m_trdCounting.joinable() || !this->m_bRunTrd) {
103 return; //maybe it was closed already?
104 }
105
106 this->m_bRunTrd = false;
107 this->m_trdCounting.join();
108 }
109
110 /// <summary>
111 /// Resets the timer's tick to 0
112 /// </summary>
113 inline void resetTick(void) noexcept {
114 this->m_ullTicks = 0;
115 }
116
117 /// <summary>
118 /// Sets the new tick value instead of the current
119 /// </summary>
120 /// <param name="tick">The tick value to set instance's tick value to</param>
121 inline void setTick(const ullint tick) noexcept {
122 this->m_ullTicks = tick;
123 }
124
125 /// <summary>
126 /// Sets the new delay instead of the current
127 /// @note It stops the thread and starts it again -- don't expect the counting to start immediately
128 /// </summary>
129 /// <param name="tps">The amount of ticks per second to set the delay to</param>
130 inline void setDelay(const double tps) {
131 this->stopThread();
132 this->m_rDelay.setFps(tps);
133 this->startThread();
134 }
135
136 /// <summary>
137 /// Returns the current tick of the timer.
138 /// @note If thread is not started/working, the return value will be the same
139 /// </summary>
140 /// <returns>ullint of the current timer tick</returns>
141 inline ullint getTick(void) const noexcept {
142 return this->m_ullTicks;
143 }
144
145 /// <summary>
146 /// Calculates the approximate world time of the timer in seconds (from the ticks)
147 /// @note If thread is not started/working, the return value will be the same
148 /// </summary>
149 /// <returns>double of the approximate world time the timer has counted (using it's ticks)</returns>
150 inline long double getWorldTime(void) const noexcept {
151 return this->m_ullTicks.load() * (long double) this->m_rDelay.getDelay();
152 }
153
154 /// <summary>
155 /// Returns the fps goal of AEFrame instance in the AETimer
156 /// @see AEFrame::getFrameRate()
157 /// </summary>
158 /// <returns>Rounded int of the approximated fps goal</returns>
159 inline double getFrameRate(void) const noexcept {
160 return this->m_rDelay.getFrameRate();
161 }
162
163 /// <summary>
164 /// Returns the maximum AEFrame's instance in the AETimer delay from the set fps, in seconds
165 /// @see AEFrame::getDelay()
166 /// </summary>
167 /// <returns>double of the maximum AEFrame's instance in the AETimer delay in real-world seconds</returns>
168 inline double getDelay(void) const noexcept {
169 return this->m_rDelay.getDelay();
170 }
171
172 /// <summary>
173 /// Stops and sleeps the thread untill the given timer hits the certain tick number.
174 /// </summary>
175 /// <param name="timer">The instance of the AETimer to wait for</param>
176 /// <param name="tick">The AETimer instance tick value to wait for</param>
177 static inline void waitForTick(const AETimer& timer, const ullint tick) noexcept {
178 AEFrame fr(timer.getFrameRate()*2);
179 while (tick > timer.getTick()){
180 fr.sleep();
181 }
182 }
183
184 /// <summary>
185 /// Stops and sleeps the thread untill the given timer goes through certain tick amount
186 /// </summary>
187 /// <param name="timer">The instance of the AETimer to wait for</param>
188 /// <param name="tick">The amount of ticks to wait for in the AETimer instance</param>
189 static inline void waitTicks(const AETimer& timer, const ullint tick) noexcept {
190 AETimer::waitForTick(timer, tick + timer.m_ullTicks);
191 }
192
193
194private:
195
196 /// <summary>
197 /// The function that does the tick counting for the timer
198 /// </summary>
199 inline void tickTimer(void) noexcept {
200 //load the atomic in the loop as fast as possible
201 //we don't care about a lot of sync, since...it's 1 byte
202 //and we just need approximate time to stop the timer
203 while (this->m_bRunTrd.load(std::memory_order_relaxed)) {
204 this->m_rDelay.sleep();
205 this->m_ullTicks++;
206
207 }
208 }
209
210 /// The frame-rater of the timer for proper delay in the tick-counting function loop
211 AEFrame m_rDelay;
212 /// The thread that does the tick counting
213 std::thread m_trdCounting;
214 /// The tick count variable
215 std::atomic<ullint> m_ullTicks;
216 /// The flag to continue running the tick counting loop
217 std::atomic<bool> m_bRunTrd;
218
219 //aaaand yeah, register the class
220 REGISTER_MODULE(AETimer);
221};
