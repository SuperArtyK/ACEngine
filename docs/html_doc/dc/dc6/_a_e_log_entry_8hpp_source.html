168 /// <summary>
169 /// Allocates the queue of the given size on the heap and returns the pointer to it's first node.
170 /// Optionally may loop the newly-allocated queue to the old queue.
171 /// @note You should delete[] the pointer after you're done using it (unless you like mem-leaks)
172 /// @note If the amt is 0, throws the std::runtime exception
173 /// </summary>
174 /// <param name="amt">The amount of entries in the queue(size)</param>
175 /// <param name="loopQueue">The flag to indicate whether to loop the generated queue at all</param>
176 /// <param name="oldqueue">The pointer to the old queue to loop the new queue to.</param>
177 /// <returns>Pointer to the first node of the allocated queue</returns>
178 static inline AELogEntry* makeQueue(const std::size_t amt, const bool loopQueue = true, AELogEntry * oldqueue = nullptr) {
179
180 //assert that amt is not 0. This is never supposed to trigger
181 //Otherwise, we're in for a hell of a ride
182 aceAssert(amt != 0, "The queue size to allocate is 0!");
183
184 //allocate new log entry list
185 AELogEntry* leptr = new AELogEntry[amt]{};
186 for (std::size_t i = 0; i < amt - 1; i++) {
187 leptr[i].m_pNextNode = leptr + i + 1; //set the next node pointers for our linked list
188 }
189 //loop the last log entry to the beginning
190
191 if (loopQueue) {
192 if (oldqueue) {
193 leptr[amt - 1].m_pNextNode = oldqueue; //beginning of old queue
194 }
195 else {
196 leptr[amt - 1].m_pNextNode = leptr; //beginning of itself
197 }
198 }
199
200 return leptr;
201 }

277 /// <summary>
278 /// Parses a given string as a log entry, and write it into passed entry object
279 /// @note If the string is less than AELE_FORMAT_MIN_SIZE or more than AELE_FORMAT_MAX_SIZE, fails the check with AELE_ERR_INVALID_LENGTH
280 /// </summary>
281 /// <param name="entry">The log entry object to write data to</param>
282 /// <param name="entryString">The string to parse</param>
283 /// <param name="flags">The flags for parsing. Refer to AELE_PARSE_STRING_* (like AELE_PARSE_STRING_JUST_VALIDATE)</param>
284 /// <returns>AELE_ERR_NOERROR (0) on success, other AELE error flags otherwise</returns>
285 static inline cint parseStringEntry(AELogEntry& entry, const std::string_view entryString, const cint flags = AELE_PARSE_STRING_FULL) {
286
287 constexpr std::size_t POS_TYPE = 22, POS_MNAME = 39;
288 if (entryString.size() > AELE_FORMAT_MAX_SIZE || entryString.size() < AELE_FORMAT_MIN_SIZE) { // it is more then the max size or less than possible size -- invalid or...somehow modified
289 return AELE_ERR_INVALID_LENGTH;
290 }
291
292 char logmessage[AELE_MESSAGE_DEBUG_SIZE+1]{}; //log message, exactly 518 characters, you know, in case it's a debug string
293 char mname[AELE_MODULENAME_SIZE + 2]{}; // +1 more character to see if it's not correct format
294 char logType[15]{}; //14 characters guaranteed
295
296 char scanfFormat[16]{}; //the format string for the sscanf. Since (sigh) macros dont evaluate while in preprocessor stage
297
298 //check the log entry's time
299 const std::time_t entryTime = ace::utils::stringToDate(entryString.data(), "[%Y-%m-%d.%X");
300 if (entryTime == -1) {
301 return AELE_ERR_INVALID_TIME;
302 }
303
304 // check for the "] " between the time and type
305 if (std::memcmp(entryString.data() + POS_TYPE - 2, "] ", 2)) {
306 return AELE_ERR_INVALID_FORMAT;
307 }
308
309 // read the possible type
310 if (std::sscanf(entryString.data() + POS_TYPE, "[%14[^]] ", logType) != 1) {
311 return AELE_ERR_INVALID_TYPE; //oops!
312 }
313
314 // check for the valid log entry's type
315 const cint entryType = AELogEntry::stringToType(logType);
316 if (entryType == -1) {
317 return AELE_ERR_INVALID_TYPE; //oops!
318 }
319
320
321 // check for the "] " between the type and module name
322 if (std::memcmp(entryString.data() + POS_MNAME - 2, "] ", 2)) {
323 return AELE_ERR_INVALID_FORMAT;
324 }
325
326 // read the possible module name
327 std::snprintf(scanfFormat, sizeof(scanfFormat), "[%%%llu[^]]", ullint(AELE_MODULENAME_SIZE + 1));
328 if (std::sscanf(entryString.data() + POS_MNAME, scanfFormat, mname) != 1) {
329 return AELE_ERR_INVALID_MNAME;
330 }
331
332 // check for the correct module name
333 const std::string_view strvMname(mname);
334 if (strvMname.size() > 31 || !ace::utils::isAlNumUs(strvMname)) {
335 return AELE_ERR_INVALID_MNAME;
336 }
337
338 //okay. we might have read that.
339 //Since the module name is variable and cannot be predicted
340 //We might have read untill the end of the "entry"
341 //so check it!
342 if (entryString.size() < (POS_MNAME + strvMname.size() + 5)) { //position of module name (39)+ size of the mname + 5 characters forward (leading [, trailing "]: ", and 1 character for message)
343 return AELE_ERR_INVALID_LENGTH;
344 }
345
346 //cool, now onwards to text
347 //check for the existing boundary between module name and message
348 if (std::memcmp(entryString.data() + POS_MNAME + strvMname.size() + 1, "]: ", 3)) {
349 return AELE_ERR_INVALID_FORMAT; //oops
350 }
351
352 if (flags == AELE_PARSE_STRING_JUST_VALIDATE) {
353 return AELE_ERR_NOERROR;
354 }
355
356 //time to write stuff
357 AELogEntry::clearEntry(entry);
358
359 if (flags & AELE_PARSE_STRING_MESSG) {
360 //cool, passed. now read untill the end
361 //the newline and or the size is guaranteed to exist
362 std::snprintf(scanfFormat, sizeof(scanfFormat), "%%%llu[^\n]", AELE_MESSAGE_DEBUG_SIZE);
363 std::sscanf(entryString.data() + POS_MNAME + strvMname.size() + 4, scanfFormat, logmessage);
364 // what if the entry is a debug one?
365 if (entryType == AELOG_TYPE_DEBUG && std::strstr(logmessage, "DEBUG->") != nullptr) {
366 std::memcpy(entry.m_sLogMessage, logmessage + sizeof("DEBUG->") - 1, AELE_MESSAGE_SIZE - sizeof("DEBUG->") + 1);
367 }
368 else {
369 std::memcpy(entry.m_sLogMessage, logmessage, AELE_MESSAGE_SIZE);
370 }
371
372 }
373
374 if (flags & AELE_PARSE_STRING_MNAME) {
375 std::memcpy(entry.m_sModuleName, mname, strvMname.size()); //module name
376 }
377 if (flags & AELE_PARSE_STRING_TIME) {
378 entry.m_tmLogTime = entryTime; //time
379 }
380 if (flags & AELE_PARSE_STRING_TYPE) {
381 entry.m_cLogType = entryType; //type
382 }
383
384
385 if (flags & AELE_PARSE_STRING_FULL) {
386 entry.m_cStatus = AELE_STATUS_READY; //status
387 }
388
389 return AELE_ERR_NOERROR;
390 }
