159 static inline AELogEntry* makeQueue(const std::size_t amt, AELogEntry* oldqueue = nullptr, const bool loopQueue = true) {
160
161 //assert that amt is not 0. This is never supposed to trigger
162 //Otherwise, we're in for a hell of a ride
163 aceAssert(amt != 0, "The queue size to allocate is 0!");
164
165 //allocate new log entry list
166 AELogEntry* leptr = new AELogEntry[amt]{};
167 for (std::size_t i = 0; i < amt-1; i++) {
168 leptr[i].m_pNextNode = leptr + i + 1; //set the next node pointers for our linked list
169 }
170 //loop the last log entry to the beginning
171
172 if (loopQueue) {
173 if (oldqueue) {
174 leptr[amt - 1].m_pNextNode = oldqueue; //beginning of old queue
175 }
176 else {
177 leptr[amt - 1].m_pNextNode = leptr; //beginning of itself
178 }
179 }
180
181 return leptr;
182 }

260 static inline cint parseStringEntry(AELogEntry& entry, const std::string_view entryString, const cint flags = AELE_PARSE_STRING_FULL) {
261
262 constexpr std::size_t POS_TYPE = 22, POS_MNAME = 39;
263 if (entryString.size() > AELE_FORMAT_MAX_SIZE || entryString.size() < AELE_FORMAT_MIN_SIZE) { // it is more then the max size or less than possible size -- invalid or...somehow modified
264 return AELE_ERR_INVALID_LENGTH;
265 }
266
267 char logmessage[AELE_MESSAGE_SIZE + sizeof("DEBUG->")]{}; //log message, exactly 518 characters, you know, in case it's a debug string
268 char mname[AELE_MODULENAME_SIZE + 2]{}; // +1 more character to see if it's not correct format
269 char logType[15]{}; //14 characters guaranteed
270
271 //check the log entry's time
272 const std::time_t entryTime = ace::utils::stringToDate(entryString.data(), "[%Y-%m-%d.%X");
273 if (entryTime == -1) {
274 return AELE_ERR_INVALID_TIME;
275 }
276
277 // check for the "] " between the time and type
278 if (std::memcmp(entryString.data() + POS_TYPE - 2, "] ", 2)) {
279 return AELE_ERR_INVALID_FORMAT;
280 }
281
282 // read the possible type
283 if (sscanf(entryString.data() + POS_TYPE, "[%14[^]] ", logType) != 1) {
284 return AELE_ERR_INVALID_TYPE; //oops!
285 }
286
287 // check for the valid log entry's type
288 const cint entryType = AELogEntry::stringToType(logType);
289 if (entryType == -1) {
290 return AELE_ERR_INVALID_TYPE; //oops!
291 }
292
293
294 // check for the "] " between the type and module name
295 if (std::memcmp(entryString.data() + POS_MNAME - 2, "] ", 2)) {
296 return AELE_ERR_INVALID_FORMAT;
297 }
298
299 // read the possible module name
300 if (sscanf(entryString.data() + POS_MNAME, "[%32[^]] ", mname) != 1) {
301 return AELE_ERR_INVALID_MNAME;
302 }
303
304 // check for the correct module name
305 const std::string_view strvMname(mname);
306 if (strvMname.size() > 31 || !ace::utils::isAlNumUs(strvMname)) {
307 return AELE_ERR_INVALID_MNAME;
308 }
309
310 //okay. we might have read that.
311 //Since the module name is variable and cannot be predicted
312 //We might have read untill the end of the "entry"
313 //so check it!
314 if (entryString.size() < (POS_MNAME + strvMname.size() + 5)) { //position of module name (39)+ size of the mname + 5 characters forward (leading [, trailing "]: ", and 1 character for message)
315 return AELE_ERR_INVALID_LENGTH;
316 }
317
318 //cool, now onwards to text
319 //check for the existing boundary between module name and message
320 if (std::memcmp(entryString.data() + POS_MNAME + strvMname.size() + 1, "]: ", 3)) {
321 return AELE_ERR_INVALID_FORMAT; //oops
322 }
323
324 if (flags == AELE_PARSE_STRING_JUST_VALIDATE) {
325 return AELE_ERR_NOERROR;
326 }
327
328 //time to write stuff
329 AELogEntry::clearEntry(entry);
330
331 if (flags & AELE_PARSE_STRING_WRITE_MESSG) {
332 //cool, passed. now read untill the end
333 //the newline and or the size is guaranteed to exist
334 sscanf(entryString.data() + POS_MNAME + strvMname.size() + 4, "%518[^\n]", logmessage);
335 // what if the entry is a debug one?
336 if (entryType == AELOG_TYPE_DEBUG && std::strstr(logmessage, "DEBUG->") != nullptr) {
337 std::memcpy(entry.m_sLogMessage, logmessage + sizeof("DEBUG->") - 1, AELE_MESSAGE_SIZE - sizeof("DEBUG->") + 1);
338 }
339 else {
340 std::memcpy(entry.m_sLogMessage, logmessage, AELE_MESSAGE_SIZE);
341 }
342
343 }
344
345 if (flags & AELE_PARSE_STRING_WRITE_MNAME) {
346 std::memcpy(entry.m_sModuleName, mname, strvMname.size()); //module name
347 }
348 if (flags & AELE_PARSE_STRING_WRITE_TIME) {
349 entry.m_tmLogTime = entryTime; //time
350 }
351 if (flags & AELE_PARSE_STRING_WRITE_TYPE) {
352 entry.m_cLogType = entryType; //type
353 }
354
355
356 if (flags & AELE_PARSE_STRING_FULL) {
357 entry.m_cStatus = AELE_STATUS_READY; //status
358 }
359
360 return AELE_ERR_NOERROR;
361 }
