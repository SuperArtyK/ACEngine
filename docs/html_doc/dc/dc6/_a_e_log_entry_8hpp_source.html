155 static inline AELogEntry* makeQueue(const std::size_t amt, AELogEntry* oldqueue = nullptr, const bool loopQueue = true) {
156
157 //assert that amt is not 0. This is never supposed to trigger
158 //Otherwise, we're in for a hell of a ride
159 aceAssert(amt != 0, "The queue size to allocate is 0!");
160
161 //allocate new log entry list
162 AELogEntry* leptr = new AELogEntry[amt]{};
163 for (std::size_t i = 0; i < amt - 1; i++) {
164 leptr[i].m_pNextNode = leptr + i + 1; //set the next node pointers for our linked list
165 }
166 //loop the last log entry to the beginning
167
168 if (loopQueue) {
169 if (oldqueue) {
170 leptr[amt - 1].m_pNextNode = oldqueue; //beginning of old queue
171 }
172 else {
173 leptr[amt - 1].m_pNextNode = leptr; //beginning of itself
174 }
175 }
176
177 return leptr;
178 }

256 static inline cint parseStringEntry(AELogEntry& entry, const std::string_view entryString, const cint flags = AELE_PARSE_STRING_FULL) {
257
258 constexpr std::size_t POS_TYPE = 22, POS_MNAME = 39;
259 if (entryString.size() > AELE_FORMAT_MAX_SIZE || entryString.size() < AELE_FORMAT_MIN_SIZE) { // it is more then the max size or less than possible size -- invalid or...somehow modified
260 return AELE_ERR_INVALID_LENGTH;
261 }
262
263 char logmessage[AELE_MESSAGE_SIZE + sizeof("DEBUG->")]{}; //log message, exactly 518 characters, you know, in case it's a debug string
264 char mname[AELE_MODULENAME_SIZE + 2]{}; // +1 more character to see if it's not correct format
265 char logType[15]{}; //14 characters guaranteed
266
267 //check the log entry's time
268 const std::time_t entryTime = ace::utils::stringToDate(entryString.data(), "[%Y-%m-%d.%X");
269 if (entryTime == -1) {
270 return AELE_ERR_INVALID_TIME;
271 }
272
273 // check for the "] " between the time and type
274 if (std::memcmp(entryString.data() + POS_TYPE - 2, "] ", 2)) {
275 return AELE_ERR_INVALID_FORMAT;
276 }
277
278 // read the possible type
279 if (sscanf(entryString.data() + POS_TYPE, "[%14[^]] ", logType) != 1) {
280 return AELE_ERR_INVALID_TYPE; //oops!
281 }
282
283 // check for the valid log entry's type
284 const cint entryType = AELogEntry::stringToType(logType);
285 if (entryType == -1) {
286 return AELE_ERR_INVALID_TYPE; //oops!
287 }
288
289
290 // check for the "] " between the type and module name
291 if (std::memcmp(entryString.data() + POS_MNAME - 2, "] ", 2)) {
292 return AELE_ERR_INVALID_FORMAT;
293 }
294
295 // read the possible module name
296 if (sscanf(entryString.data() + POS_MNAME, "[%32[^]] ", mname) != 1) {
297 return AELE_ERR_INVALID_MNAME;
298 }
299
300 // check for the correct module name
301 const std::string_view strvMname(mname);
302 if (strvMname.size() > 31 || !ace::utils::isAlNumUs(strvMname)) {
303 return AELE_ERR_INVALID_MNAME;
304 }
305
306 //okay. we might have read that.
307 //Since the module name is variable and cannot be predicted
308 //We might have read untill the end of the "entry"
309 //so check it!
310 if (entryString.size() < (POS_MNAME + strvMname.size() + 5)) { //position of module name (39)+ size of the mname + 5 characters forward (leading [, trailing "]: ", and 1 character for message)
311 return AELE_ERR_INVALID_LENGTH;
312 }
313
314 //cool, now onwards to text
315 //check for the existing boundary between module name and message
316 if (std::memcmp(entryString.data() + POS_MNAME + strvMname.size() + 1, "]: ", 3)) {
317 return AELE_ERR_INVALID_FORMAT; //oops
318 }
319
320 if (flags == AELE_PARSE_STRING_JUST_VALIDATE) {
321 return AELE_ERR_NOERROR;
322 }
323
324 //time to write stuff
325 AELogEntry::clearEntry(entry);
326
327 if (flags & AELE_PARSE_STRING_MESSG) {
328 //cool, passed. now read untill the end
329 //the newline and or the size is guaranteed to exist
330 sscanf(entryString.data() + POS_MNAME + strvMname.size() + 4, "%518[^\n]", logmessage);
331 // what if the entry is a debug one?
332 if (entryType == AELOG_TYPE_DEBUG && std::strstr(logmessage, "DEBUG->") != nullptr) {
333 std::memcpy(entry.m_sLogMessage, logmessage + sizeof("DEBUG->") - 1, AELE_MESSAGE_SIZE - sizeof("DEBUG->") + 1);
334 }
335 else {
336 std::memcpy(entry.m_sLogMessage, logmessage, AELE_MESSAGE_SIZE);
337 }
338
339 }
340
341 if (flags & AELE_PARSE_STRING_MNAME) {
342 std::memcpy(entry.m_sModuleName, mname, strvMname.size()); //module name
343 }
344 if (flags & AELE_PARSE_STRING_TIME) {
345 entry.m_tmLogTime = entryTime; //time
346 }
347 if (flags & AELE_PARSE_STRING_TYPE) {
348 entry.m_cLogType = entryType; //type
349 }
350
351
352 if (flags & AELE_PARSE_STRING_FULL) {
353 entry.m_cStatus = AELE_STATUS_READY; //status
354 }
355
356 return AELE_ERR_NOERROR;
357 }
