37/// <summary>
38/// This module is the engine's **precise frame/loop-cycle limiter**.
39///
40/// It allows to limit the execution of a loop to a set amount of iterations per second. <br>
41/// Or, instead have the set delay per each loop cycle.
42///
43/// It works by sleeping the difference between the time spent by code in the loop and the time per each frame.
44///
45/// For example: <br>
46/// If you want the loop to have 100 cycles/second or fps (10 milliseconds per cycle),
47/// and the code takes **4ms** to complete, then the AEFrame will sleep the difference. <br>
48/// In this case it's **6ms** to make it up to the 10ms per cycle.
49///
50/// @note By default it uses the value of #ENGINE_FPS as the fps target for the loops
51/// </summary>
52class AEFrame : public __AEModuleBase<AEFrame> {
53
54public:
55
56 /// This is an an **alias for the internal clock** of the AEFrame module.
57 /// @warning The resulution of the AEFrameClock that your compiler gives
58 /// must be at least 1 microsecond, otherwise it won't compile (breaks assertion).
59 using AEFrameClock = AEF_CLOCK;
60
61//constructors
62 /// <summary>
63 /// **Class constructor** -- constructs the instance with the **fps as the delay**.
64 ///
65 /// @see AEFrame::setFPS()
66 /// @attention If you pass it negative, 0, or more than 1 million, it will disable the
67 /// delay (sets AEFrame::m_bNoDelay to **true**)
68 /// @note Defaults the fps delay to #ENGINE_FPS
69 /// </summary>
70 /// <param name="fps">The fps goal that the frame delay module will try to achieve</param>
71 explicit AEFrame(const double fps = ENGINE_FPS) noexcept :
72 m_tdChronoDelay((fps <= 0 || fps > 1000000.0) ? microSec(0) : microSec(llint(1000000.0 / fps))),
73 m_tpTimepoint(AEFrameClock::now()), m_fFPS(fps), m_bNoDelay((fps <= 0 || fps > 1000000.0))
74 {}
75
76 /// <summary>
77 /// **Copy constructor** -- copies data from a passed AEFrame instance.
78 /// </summary>
79 /// <param name="two">The second AEFrame instance</param>
80 AEFrame(const AEFrame& two) noexcept :
81 //COPYCONSTRUCT_MODULE(AEFrame, two),
82 m_tdChronoDelay(two.m_tdChronoDelay),
83 m_tpTimepoint(AEFrameClock::now()), m_fFPS(two.m_fFPS), m_bNoDelay(two.m_bNoDelay)
84 {}
85
86 /// <summary>
87 /// **Copy-assignment operator** -- copies data from a passed AEFrame instance.
88 ///
89 /// @note This resets the internal time point (calls AEFrame::resetTimePoint())
90 /// </summary>
91 /// <param name="two">The second AEFrame instance</param>
92 /// <returns>
93 /// The reference to the resulting AEFrame instance.
94 /// </returns>
95 AEFrame& operator=(const AEFrame& two) noexcept {
96 this->m_tdChronoDelay = two.m_tdChronoDelay;
97 this->m_tpTimepoint = AEFrameClock::now();
98 this->m_fFPS = two.m_fFPS;
99 this->m_bNoDelay = two.m_bNoDelay;
100 this->resetTimePoint();
101
102 return *this;
103 }
104

106//utils
107 /// <summary>
108 /// Sets the delay of the instance as the **frames (loop cycles) per second** to reach in execution (fps goal).
109 ///
110 /// @attention If you pass it negative, 0, or more than 1 million, it will disable the
111 /// delay (sets AEFrame::m_bNoDelay to **true**)
112 /// @note This resets the internal time point (calls AEFrame::resetTimePoint())
113 /// </summary>
114 /// <param name="fps">The frames(cycles) per second to set the delay as</param>
115 inline void setFps(const double fps) noexcept {
116 if (fps <= 0.0 || fps > 1000000.0) { this->m_bNoDelay = true; return; }
117 this->m_tdChronoDelay = microSec(llint(1000000.0 / fps));
118 this->m_tpTimepoint = AEFrameClock::now();
119 this->m_fFPS = fps;
120 this->m_bNoDelay = false;
121 }
122
123 /// <summary>
124 /// Sets the delay of the instance as **maximum time for each loop cycle**.
125 ///
126 /// @attention If you pass negative or 0, it will disable the delay (sets AEFrame::m_bNoDelay to **true**)
127 /// </summary>
128 /// <param name="sec">The delay of instance/time for each loop cycle</param>
129 inline void setDelay(const double sec) {
130 if (sec <= 0.0) { this->m_bNoDelay = true; return; }
131 this->m_tdChronoDelay = microSec(llint(1000000.0 * sec));
132 this->m_tpTimepoint = AEFrameClock::now();
133 this->m_fFPS = 1.0 / sec;
134 this->m_bNoDelay = false;
135 }
136
137 /// <summary>
138 /// Puts the current thread to **sleep for the delay that was set earlier**.
139 ///
140 /// Example: if the fps delay was set to 30fps (~~33ms delay in real time) and a game
141 /// loop takes just 13ms, then the thread will sleep the rest of 20ms to make the
142 /// game loop run at around ~~33ms of delay or around 30fps.
143 ///
144 /// @attention If the delay is disabled (AEFrame::m_bNoDelay is **true**), just returns and doesn't do anything
145 /// @note If the AEFrame::m_tpTimepoint gets too out of touch with real time, resets the time point (calls AEFrame::resetTimePoint())
146 /// The definition of "out of touch" is controlled by #AEF_RESET_TIME_WHEN_BEHIND_MS.
147 /// @note Automatic timepoint resetting works **only** if #AEF_RESET_TIME_WHEN_BEHIND_MS is **enabled**
148 /// </summary>
149 inline void sleep(void) noexcept {
150 // add to time point
151 if (this->m_bNoDelay) {
152 return;
153 }
154#if AEF_RESET_TIME_WHEN_BEHIND_MS > 0
155 if ((AEFrameClock::now() - this->m_tpTimepoint) > m_tdResetTime) {
156 this->resetTimePoint();
157 }
158#endif
159
160 this->m_tpTimepoint += this->m_tdChronoDelay;
161 // and sleep until that time point
162 sleepUntil(this->m_tpTimepoint);
163 }
164
165 /// <summary>
166 /// **Resets the time point** in the AEFrame to the real (current) time.
167 /// Helps if the AEFrame::m_tpTimepoint time point drifted far from the reality.
168 /// </summary>
169 inline void resetTimePoint(void) noexcept {
170 this->m_tpTimepoint = AEFrameClock::now();
171 }
172
173 /// <summary>
174 /// Returns the **fps goal of the instance**.
175 /// </summary>
176 /// <returns>
177 /// The fps goal of the instance as type **double**.
178 /// </returns>
179 inline double getFrameRate(void) const noexcept {
180 return this->m_fFPS;
181 }
182
183 /// <summary>
184 /// Returns the **maximum loop delay in the instance** in seconds
185 ///
186 /// Or better: the time, in seconds, for each loop cycle that AEFrame::sleep() is called
187 /// </summary>
188 /// <returns>
189 /// The delay of the instance as seconds as type **double**.
190 /// </returns>
191 inline double getDelay(void) const noexcept {
192 return this->m_tdChronoDelay.count();
193 }
194
195private:
196
197 static_assert(
198 (std::chrono::duration_cast<std::chrono::microseconds>(typename AEFrameClock::duration(1)).count()) <= 1,
199 "Your C++ compiler doesn't provide a precise enough steady_clock."
200 );
201
202
203 /// The **time duration for resetting the AEFrame::m_tpTimepoint** if it gets too different with real time.
204 /// Only exists if #AEF_RESET_TIME_WHEN_BEHIND_MS is defined
205 /// @see #AEF_RESET_TIME_WHEN_BEHIND_MS
206#if AEF_RESET_TIME_WHEN_BEHIND_MS > 0
207 static constexpr timeDur<double> m_tdResetTime = microSec(milliSec(llint(AEF_RESET_TIME_WHEN_BEHIND_MS)));
208#endif
209
210 /// The time duration for the **sleeping delay**
211 timeDur<double> m_tdChronoDelay;
212 /// The timepoint, that sets time **when to wake up the thread**.
213 timePoint<AEFrameClock, decltype(m_tdChronoDelay)> m_tpTimepoint;
214 double m_fFPS;
215 /// The flag if **we don't need the delay**, err, for some reason.
216 bool m_bNoDelay;
217
218 //aaand we have to register it too
219 REGISTER_MODULE(AEFrame);
220};
