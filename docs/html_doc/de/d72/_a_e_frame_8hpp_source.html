27/// <summary>
28/// @brief This is the ArtyK's Engine's frame delay module -- it allows to create precise delays for game loops.
29/// It works by sleeping the difference between the time spent by code in the loop and the time per each frame.
30///
31/// For example: if you want the loop to have 100 cycles/second or fps (10 milliseconds), and the code takes 5ms to complete,
32/// then the AEFrame will sleep that loop thread for 5ms.
33///
34/// Hungarian notation is fd. (m_fdMyFrameDelay)
35/// @note To get this to work properly, make an object outside of the loop, and put the call to AEFrame::sleep() as the very last/first thing in the loop.
36/// @warning If the game loop takes more time than the set fps goal amount, the frame times can be unpredictable (and...delay might also not work) (working to fix that)
37/// </summary>
38class AEFrame : public __AEModuleBase<AEFrame> {
39
40public:
41
42//constructors
43 /// <summary>
44 /// Class constructor.
45 /// @note if you pass it 0 or negative number it will disable the delay untill a proper value is given
46 /// </summary>
47 /// <param name="fps">The fps goal that the frame delay module will try to achieve</param>
48 explicit AEFrame(const double fps = ENGINE_FPS) noexcept :
49 m_tdChronoDelay((fps <= 0) ? microSec(0) : microSec(ullint(1000000.0 / fps))),
50 m_tpTimepoint(getSteadyTime()), m_fFPS(fps), m_bNoDelay((fps <= 0))
51 {}
52
53 AEFrame(const AEFrame& two) noexcept :
54 COPYCONSTRUCT_MODULE(AEFrame, two), m_tdChronoDelay(two.m_tdChronoDelay),
55 m_tpTimepoint(getSteadyTime()), m_fFPS(two.m_fFPS), m_bNoDelay(two.m_bNoDelay)
56 {}
57
58 AEFrame& operator=(const AEFrame& two) noexcept {
59 this->m_tdChronoDelay = two.m_tdChronoDelay;
60 this->m_tpTimepoint = getSteadyTime();
61 this->m_fFPS = two.m_fFPS;
62 this->m_bNoDelay = two.m_bNoDelay;
63 this->resetTimePoint();
64
65 return *this;
66 }
67
68
69//utils
70 /// <summary>
71 /// Resets the delay value to what you pass to it.
72 /// @note if you pass it 0 or negative number it will disable the delay untill a proper value is given
73 /// </summary>
74 /// <param name="fps">The fps goal that the frame delay module will try to achieve</param>
75 inline void setFps(const double fps) noexcept {
76 if (fps <= 0) { this->m_bNoDelay = true; return; }
77 this->m_tdChronoDelay = microSec(ullint(1000000.0 / fps));
78 this->m_tpTimepoint = getSteadyTime();
79 this->m_fFPS = fps;
80 }
81
82 /// <summary>
83 /// Puts the current thread to sleep up to the set fps goal.
84 /// Example: if the fps goal was set to 30fps (~~33ms) and a game loop takes 13ms,
85 /// then the thread will sleep the rest of 20ms to make the game loop run at 30fps.
86 /// </summary>
87 inline void sleep(void) noexcept {
88 // add to time point
89 if (this->m_bNoDelay) {
90 return;
91 }
92
93 this->m_tpTimepoint += this->m_tdChronoDelay;
94 // and sleep until that time point
95 sleepUntil(this->m_tpTimepoint);
96 }
97
98 /// <summary>
99 /// Resets the time point in the AEFrame to current system time.
100 /// Helps if the gameloop delay was much bigger than AEFrames for a long time (it causes delay to stop working untill it catches up)
101 /// </summary>
102 inline void resetTimePoint(void) noexcept {
103 this->m_tpTimepoint = getSteadyTime();
104 }
105
106 /// <summary>
107 /// Returns the fps goal of AEFrame instance.
108 /// </summary>
109 /// <returns>double of the given fps goal</returns>
110 inline double getFrameRate(void) const noexcept { return this->m_fFPS; }
111
112 /// <summary>
113 /// Returns the maximum AEFrame's delay from the set fps, in seconds.
114 /// </summary>
115 /// <returns>double of the maximum AEFrame's delay in real-world seconds</returns>
116 inline double getDelay(void) const noexcept { return this->m_tdChronoDelay.count(); }
117
118private:
119
120 /// The time duration for the delay in sleeps
121 timeDur<double> m_tdChronoDelay;
122 /// The timepoint, that sets time when to wake up the thread
123 timePoint<SteadyClock, decltype(m_tdChronoDelay)> m_tpTimepoint;
124 double m_fFPS;
125 /// The flag if we don't need the delay, err, for some reason
126 bool m_bNoDelay;
127
128 //aaand we have to register it too
129 REGISTER_MODULE(AEFrame)
130};
