170 constexpr bool isNum(const char* const strnum) noexcept {
171 //null pointer check
172 if (!strnum) {
173 return false;
174 }
175
176 const std::size_t len = std::strlen(strnum);
177
178 //is it 0?
179 if (len == 0) {
180 return false;
181 }
182 size_t i = 0;
183 if (strnum[0] == '-') {
184 //check if string has only the minus in it
185 if (len < 2) {
186 return false;
187 }
188
189 i = 1;
190 }
191 //quick check, if string has 0 as the first number...hey that's invalid!
192 if (strnum[i] == '0' && (i == 1 || len > 1)) {
193 return false;
194 }
195 for (; i < len; i++) {
196 if (strnum[i] < '0' || strnum[i] > '9') {
197 return false;
198 }
199 }
200 //passed
201 return true;
202 }
