159 constexpr bool isNum(const char* strnum) noexcept {
160 //null pointer check
161 if (!strnum) {
162 return false;
163 }
164
165 const std::size_t len = std::strlen(strnum);
166
167 //is it 0?
168 if (len == 0) {
169 return false;
170 }
171 size_t i = 0;
172 if (strnum[0] == '-') {
173 //check if string has only the minus in it
174 if (len < 2) {
175 return false;
176 }
177
178 i = 1;
179 }
180 //quick check, if string has 0 as the first number...hey that's invalid!
181 if (strnum[i] == '0' && (i == 1 || len > 1)) {
182 return false;
183 }
184 for (; i < len; i++) {
185 if (strnum[i] < '0' || strnum[i] > '9') {
186 return false;
187 }
188 }
189 //passed
190 return true;
191 }
