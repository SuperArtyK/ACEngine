136 constexpr bool isNum(const char* strnum) {
137 //null pointer check
138 if (!strnum) {
139 return false;
140 }
141
142 std::size_t len = std::strlen(strnum);
143
144 //is it 0?
145 if (len == 0) {
146 return false;
147 }
148 size_t i = 0;
149 if (strnum[0] == '-') {
150 //check if string has only the minus in it
151 if (len < 2) {
152 return false;
153 }
154
155 i = 1;
156 }
157 //quick check, if string has 0 as the first number...hey that's invalid!
158 if (strnum[i] == '0' && (i == 1 || len > 1)) {
159 return false;
160 }
161 for (; i < len; i++) {
162 if (strnum[i] < '0' || strnum[i] > '9') {
163 return false;
164 }
165 }
166 //passed
167 return true;
168 }
