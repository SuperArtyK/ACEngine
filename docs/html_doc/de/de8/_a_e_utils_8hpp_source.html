253 /// <summary>
254 /// Checks if the given **string is a numerical value**.
255 /// </summary>
256 /// <typeparam name="checkFloat">Template flag whether to check for integer values (false) or any/float values (true)</typeparam>
257 /// <param name="strnum">String object (any, std::string, c-string, etc), "number" to check</param>
258 /// <returns>
259 /// If **strnum** is a numeric string:
260 /// * **True**
261 ///
262 /// Otherwise, or if the size of **strnum** is 0:
263 /// * **False**
264 /// </returns>
265 template<const bool checkFloat = true>
266 constexpr bool isNum(const std::string_view strnum) noexcept {
267 const std::size_t len = strnum.size();
268 //is it empty?
269 if (len == 0) {
270 return false;
271 }
272 size_t i = 0;
273 if (strnum[0] == '-') {
274 //check if string has only the minus in it
275 if (len < 2) {
276 return false;
277 }
278 i = 1;
279 }
280 //quick check, if string has 0 as the first number, minus before it
281 //and no dot after
282 if (strnum[i] == '0' && i == 1 &&
283 (len == 2 ||
284 (len > 2 && strnum[++i] != '.'))) {
285 return false; //hey that's invalid!
286 }
287
288 bool metDecimal = false;
289 for (; i < len; i++) {
290 if (strnum[i] < '0' || strnum[i] > '9') {
291 if constexpr (checkFloat) {
292 if (strnum[i] == '.' && !metDecimal) {
293 metDecimal = true;
294 }
295 else {
296 return false;
297 }
298 }
299 else {
300 return false;
301 }
302
303 }
304 }
305 //passed
306 return true;
307 }

309 /// <summary>
310 /// Formats the current **(c)time point as YYYY-MM-DD.HH:mm:SS string**.
311 /// </summary>
312 /// <param name="timept">C time point</param>
313 /// <returns>
314 /// If the **timept** is positive:
315 /// * **std::string** containing the formatted **timept** with the set format, of size #DATETIME_STRING_SIZE.
316 ///
317 /// Otherwise (**timept** is negative):
318 /// * Emtpy **std::string**
319 /// </returns>
320 inline std::string formatDate(const std::time_t timept) {
321 if (timept < 0) {
322 return "";
323 }
324
325 struct tm tstruct {};
326#ifdef _WIN32
327 localtime_s(&tstruct, &timept);
328#else
329 tstruct = *localtime(&timept);
330#endif
331 char buff[DATETIME_STRING_SIZE]{};
332 snprintf(buff, DATETIME_STRING_SIZE, "%04d-%02d-%02d.%02d:%02d:%02d",
333 tstruct.tm_year + 1900, tstruct.tm_mon + 1,
334 tstruct.tm_mday, tstruct.tm_hour, tstruct.tm_min,
335 tstruct.tm_sec);
336 return buff;
337 }

383 /// <summary>
384 /// Converts the given c-string to a (c)time point (std::time_t), using a set format
385 /// @note Format specifiers https://en.cppreference.com/w/cpp/io/manip/get_time
386 /// </summary>
387 /// <param name="timestr">The c-string to a formatted time</param>
388 /// <param name="timeformat">A c-string of the format that it uses</param>
389 /// <returns>
390 /// If the **timestr** and **timeformat** isn't nullptr, and **timestr** follows the set format:
391 /// * Time since epoch as a **std::time_t** object
392 ///
393 /// Otherwise:
394 /// * **-1** as the **std::time_t** object
395 /// </returns>
396 inline std::time_t stringToDate(const char* const timestr, const char* const timeformat = "%Y-%m-%d.%X") {
397 if (!timestr || !timeformat) {
398 return std::time_t(-1);
399 }
400 struct tm tstruct {};
401 std::istringstream iss(timestr);
402 iss >> std::get_time(&tstruct, timeformat);
403
404 // use system timezone databases for...proper time shift;
405 // thanks Michael Paxton and Lightness-Races-in-Orbit: https://stackoverflow.com/a/55475752
406 tstruct.tm_isdst = -1;
407 return std::mktime(&tstruct);
408 }
