183 constexpr bool isNum(const char* const strnum) noexcept {
184 //null pointer check
185 if (!strnum) {
186 return false;
187 }
188
189 const std::size_t len = std::strlen(strnum);
190
191 //is it 0?
192 if (len == 0) {
193 return false;
194 }
195 size_t i = 0;
196 if (strnum[0] == '-') {
197 //check if string has only the minus in it
198 if (len < 2) {
199 return false;
200 }
201
202 i = 1;
203 }
204 //quick check, if string has 0 as the first number...hey that's invalid!
205 if (strnum[i] == '0' && (i == 1 || len > 1)) {
206 return false;
207 }
208 for (; i < len; i++) {
209 if (strnum[i] < '0' || strnum[i] > '9') {
210 return false;
211 }
212 }
213 //passed
214 return true;
215 }
