188 constexpr bool isNum(const char* const strnum) noexcept {
189 //null pointer check
190 if (!strnum) {
191 return false;
192 }
193
194 const std::size_t len = std::strlen(strnum);
195
196 //is it 0?
197 if (len == 0) {
198 return false;
199 }
200 size_t i = 0;
201 if (strnum[0] == '-') {
202 //check if string has only the minus in it
203 if (len < 2) {
204 return false;
205 }
206
207 i = 1;
208 }
209 //quick check, if string has 0 as the first number...hey that's invalid!
210 if (strnum[i] == '0' && (i == 1 || len > 1)) {
211 return false;
212 }
213 for (; i < len; i++) {
214 if (strnum[i] < '0' || strnum[i] > '9') {
215 return false;
216 }
217 }
218 //passed
219 return true;
220 }
