31// queue decrease algorithm
32// since we have the access to the maximum queue value, and it is being checked in the
33// ptrFromIndex(), we have a way to restrict the access to certain nodes for the writeToLog()
34// All by changing the max queue value to the value of the second-to-last node in the allocation vector
35// That way, when we wrap the index around next time, it wraps around the nodes that we want to keep
36// And completely ignores the last nodes we want to delete.
37// Though how to delete information from those delete-me nodes, since the writer-thread is the only thing
38// That still has access to those nodes. And we don't want to delete the unfilled ones

109 inline cint openLog(const std::string_view fname, const bool clearLog = false) {
110 if (this->isOpen()) {
111 return AEFW_ERR_FILE_ALREADY_OPEN;
112 }
113 const cint ret = this->m_fwLogger.openFile(fname, !clearLog * AEFW_FLAG_APPEND);
114 if (ret != AEFW_ERR_NOERROR) {
115 return ret;
116 }
117 this->writeToLog("Opened the log session in the file: \"" + std::string(fname) + '\"', AELOG_TYPE_OK, this->m_sModulename);
118 return this->startWriter();
119 }

128 inline cint closeLog(void) {
129 if (this->isClosed()) {
130 return AEFW_ERR_FILE_NOT_OPEN;
131 }
132 this->writeToLog("Closing the log session in the file: \"" + this->m_fwLogger.getFullFileName() + '\"', AELOG_TYPE_OK, this->m_sModulename);
133 this->stopWriter();
134 this->m_fwLogger.closeFile();
135 return AEFW_ERR_NOERROR;
136 }
