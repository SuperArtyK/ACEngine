22// queue decrease algorithm
23// since we have the access to the maximum queue value, and it is being checked in the
24// ptrFromIndex(), we have a way to restrict the access to certain nodes for the writeToLog()
25// All by changing the max queue value to the value of the second-to-last node in the allocation vector
26// That way, when we wrap the index around next time, it wraps around the nodes that we want to keep
27// And completely ignores the last nodes we want to delete.
28// Though how to delete information from those delete-me nodes, since the writer-thread is the only thing
29// That still has access to those nodes. And we don't want to delete the unfilled ones

51/// <summary>
52/// @brief The ArtyK's Engine's Logger module -- it manages the writing to the log files.
53/// A wrapper around AEFW functionality for writing, and AELogEntry for mass-formatting of entries to text
54///
55/// The log is written by requesting and filling the entry in the queue.
56/// The AELogger instance launches the separate thread that looks at the entries in the queue,
57/// retrieves and formats the data in them, and writes it to the file. Afterwards that entry in the queue is cleared.
58///
59/// Hungarian notation is lg. (m_lgMyLogger)
60/// </summary>
61/// @todo Implement copy constructors and copy assignment
62/// @todo Add the ability to open the same log file/redirect the instance requests to the one that has it open first.
63/// @bug The queue can expand if it's too little. But....I don't know how to shrink it. (working on it)

312 /// <summary>
313 /// Writes to log directly, without thinking of the queue
314 /// @warning Use it with caution, when you sure that it won't compromise the log integrity (you know, race conditions with fwrite() in the AELogger::logWriterThread())
315 /// </summary>
316 /// <param name="logmessg">Message of the log</param>
317 /// <param name="logtype">the type/severity of the log</param>
318 /// <param name="logmodule">the module name that requested the log</param>
319 inline void writeToLogDirectly(const std::string_view logmessg, const cint logtype = AELOG_TYPE_INFO, const std::string_view logmodule = AELOG_DEFAULT_MODULE_NAME) noexcept {
320 AELogEntry lEntry{
321 "", // message will be written to later
322 "", // modulename will be written to later
323 std::time(nullptr), // m_tmLogTime
324 nullptr, // m_pNextNode
325 logtype, // m_cLogType
326 //AELE_STATUS_INVALID, // m_cStatus; isn't needed since the function doesn't check it anyway
327 };
328
329 std::memcpy(lEntry.m_sLogMessage, logmessg.data(), (logmessg.size() < AELE_MESSAGE_SIZE)? logmessg.size() : AELE_MESSAGE_SIZE);
330 std::memcpy(lEntry.m_sModuleName, logmodule.data(), (logmodule.size() < AELE_MODULENAME_SIZE) ? logmodule.size() : AELE_MODULENAME_SIZE);
331 char str[AELE_FORMAT_MAX_SIZE]{};
332 //AELogEntry::formatEntry(str, lEntry);
333 this->m_fwLogger.writeData_ptr(str, AELogEntry::formatEntry(str, lEntry), 1, false);
334 this->m_fwLogger.flushFile();
335
336 }
