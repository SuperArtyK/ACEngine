56cint AELogParser::nextEntry(AELogEntry& entry, const cint severity) {
57 if (this->isClosed()) {
58 return AEFR_ERR_FILE_NOT_OPEN;
59 }
60
61 char str[AELE_FORMAT_MAX_SIZE + 2]{}; // 1 character more than the log entry - to determine the validity with size
62
63 //cycle the current entry number untill we find an entry with proper severity level
64 while (1) {
65
66 if (++this->m_ullCurrentEntry >= this->m_vecEntryIndices.size()) {
67 return AEFR_ERR_READ_EOF;
68 }
69 if (this->m_vecEntryIndices[this->m_ullCurrentEntry].second >= severity) {
70 break;
71 }
72 }
73
74 this->m_frLogReader.setCursorPos(this->m_vecEntryIndices[this->m_ullCurrentEntry].first, SEEK_SET);
75
76 const cint readret = this->m_frLogReader.readStringNL(str, sizeof(str) - 1);
77 if (readret == AEFR_ERR_NOERROR || readret == AEFR_ERR_READ_EOF) { // check even if it's EOF. It's not an error, but a warning from file reader
78 return AELogEntry::parseStringEntry(entry, str, AELE_PARSE_STRING_FULL);
79 }
80
81 return readret;
82}
