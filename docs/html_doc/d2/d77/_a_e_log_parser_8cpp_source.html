119AELogEntryInfo AELogParser::findNextEntry(const cint severity, const std::string_view mname, const bool strictSeverity) {
120
121 if (this->isClosed()) {
122 return AELogEntryInfo::invalidEntry();
123 }
124
125 if (!mname.empty() && !this->m_mapModuleNames.contains(mname.data())) {
126 return AELogEntryInfo{.mnameIndex = AELEI_INVLAID_MNAME};
127 }
128
129 if (!ace::utils::isInRange(AELOG_TYPE_DEBUG, AELOG_TYPE_FATAL_ERROR, severity)) {
130 return AELogEntryInfo{ .logType = AELEI_INVALID_TYPE };
131 }
132
133 // ternary with lambda.
134 // F to code readability
135 // basically a lambda to check severity of passed severity values
136 // strictSeverity requires the severities to be the *same*
137 // otherwise check if the passed severity is equal or higher than the "filter"
138 const auto checkSeverity = (strictSeverity) ?
139 [](const cint curEntrySeverity, const cint curSeverity) noexcept { return curEntrySeverity == curSeverity; } : // the strict, *exact* severity check
140 [](const cint curEntrySeverity, const cint curSeverity) noexcept { return curEntrySeverity >= curSeverity; }; // the normal severity check
141
142 const short mnameIndex = (!mname.empty()) ? this->m_mapModuleNames[mname.data()] : -1;
143
144 const auto checkMName = (mname.empty()) ?
145 [](const short& controlIndex, const short& mIndex) noexcept { return true; } : // if the module name is empty -- return true always (we aren't checking for it)
146 [](const short& controlIndex, const short& mIndex) noexcept { return (controlIndex == mIndex); };
147
148 while (1) {
149
150 if (++this->m_ullCurrentEntry >= this->m_vecEntryIndices.size()) {
151 return AELogEntryInfo { .cursorIndex = AELEI_INVALID_CURSOR};
152 }
153 if (checkSeverity(this->m_vecEntryIndices[this->m_ullCurrentEntry].logType, severity) && //check the severity with lambdas
154 checkMName(mnameIndex, this->m_vecEntryIndices[this->m_ullCurrentEntry].mnameIndex)) {
155
156 break;
157 }
158 }
159
160 return this->m_vecEntryIndices[this->m_ullCurrentEntry];
161
162}
