11cint AELogParser::openLog(const std::string_view fname) {
12 if (this->isOpen()) {
13 return AEFR_ERR_OPEN_FILE_ALREADY_OPENED;
14 }
15 const cint ret = this->m_frLogReader.openFile(fname);
16 if (ret != AELP_ERR_NOERROR) {
17 return ret;
18 }
19
20 // and index/parse log
21 char fileline[AELE_FORMAT_MAX_SIZE + 1]{}; //+1 more character to determine the validity)
22 cint fileRead = AEFR_ERR_NOERROR;
23 cint parsedEntry = AELP_ERR_NOERROR;
24 llint cursor = 0;
25 AELogEntry dummyentry{};
26 short mnameIndex = 0;
27
28
29 m_vecEntryIndices.reserve(AELOG_DEFAULT_QUEUE_SIZE * 10);
30 m_vecInvalidEntryIndices.reserve(AELOG_DEFAULT_QUEUE_SIZE);
31 m_mapModuleNames.reserve(AELOG_DEFAULT_QUEUE_SIZE);
32
33 while (1) {
34 cursor = this->m_frLogReader.getCursorPos();
35 fileRead = this->m_frLogReader.readStringNL(fileline, sizeof(fileline) - 1);
36
37 if (fileline[0] == '\n') { // it's an empty line -- skipping...
38 break;
39 }
40 parsedEntry = AELogEntry::parseStringEntry(dummyentry, fileline, AELE_PARSE_STRING_TYPE | AELE_PARSE_STRING_MNAME);
41
42
43 switch (parsedEntry) {
44 case AELP_ERR_NOERROR:
45 if (this->m_mapModuleNames.contains(dummyentry.m_sModuleName)) {
46 this->m_vecEntryIndices.emplace_back(cursor, this->m_mapModuleNames[dummyentry.m_sModuleName].second, dummyentry.m_cLogType);
47 }
48 else {
49 this->m_mapModuleNames[dummyentry.m_sModuleName].second = ++mnameIndex;
50 this->m_vecEntryIndices.emplace_back(cursor, mnameIndex, dummyentry.m_cLogType); // a bit of optimisation
51 }
52
53 this->m_mapModuleNames[dummyentry.m_sModuleName].first++;
54 this->m_arrEntryAmount[dummyentry.m_cLogType + 1]++;
55 break;
56
57 default:
58 if (fileRead != AEFR_ERR_NOERROR) {
59 return ret;
60 }
61 this->m_vecInvalidEntryIndices.push_back(cursor);
62 this->m_arrEntryAmount[0]++;
63 break;
64 }
65
66 }
67
68 m_vecEntryIndices.shrink_to_fit();
69 m_vecInvalidEntryIndices.shrink_to_fit();
70
71 return ret;
72}

134cint AELogParser::filterQueue(AELogEntry*& ptr, const cint severity, const bool strictSeverity, const std::string_view mname, const bool reallocQueue) {
135 // error handling
136 if (ptr == nullptr) { // nullptr was passed, lol
137 return AELP_ERR_INVALID_QUEUE;
138 }
139 // its all the sameeeee
140 if (severity == AELP_SEVERITY_ALL && mname.empty()) {
141 return AELP_ERR_INVALID_FILTER; // invalid filter option
142 }
143
144 AELogEntry* cur = ptr;
145 AELogEntry kludge{ .m_pNextNode = ptr }; // a kludge variable for the pastEntry to work
146 AELogEntry* past = &kludge;
147 std::size_t newQueueSize = 0;
148
149 const std::size_t szCheck = ace::math::min<std::size_t>(mname.size(), AELE_MODULENAME_SIZE);
150 const auto cmpMName = (mname.empty()) ?
151 ([](const char* const entryName, const std::size_t szCheck, const std::string_view& filterMname) noexcept { return true; }) : // if the module name is empty -- return true always (we aren't checking for it)
152 ([](const char* const entryName, const std::size_t szCheck, const std::string_view& filterMname) noexcept { return !std::memcmp(entryName, filterMname.data(), szCheck); });
153
154
155 //check if it even has the stuff matching the filter
156 while (cur) {
157 if (AELogParser::checkSeverity(cur->m_cLogType, severity, strictSeverity) && cmpMName(cur->m_sModuleName, szCheck, mname)) {
158 goto foundSeverity; // we found the severity, break the loop
159 break;
160 }
161
162 cur = cur->m_pNextNode;
163 }
164
165 //didn't quite catch it: error reporting
166 // there is no log entry with the given severity
167 if (mname.empty()) {
168 return AELP_ERR_INVALID_SEVERITY; // no module name filter
169 }
170 if (severity == AELP_SEVERITY_ALL) {
171 return AELP_ERR_INVALID_MODULE_NAME; // no severity filter
172 }
173 return AELP_ERR_FILTER_NO_MATCHES; //both exist, but none were captured -- no matches!
174
175 //welp, we found it
176foundSeverity:
177 while (cur) {
178 if (AELogParser::checkSeverity(cur->m_cLogType, severity, strictSeverity) && cmpMName(cur->m_sModuleName, szCheck, mname)) {
179 past = past->m_pNextNode;
180 past->copyEntry(*cur);
181 newQueueSize++;
182 }
183 cur = cur->m_pNextNode;
184
185 }
186 past->m_pNextNode = nullptr;
187
188 if (reallocQueue) {
189 //clean up the queue
190 //or...make a new, smaller queue instead!
191 cur = ptr;
192 AELogEntry* const newQueue = AELogEntry::makeQueue(newQueueSize, false);
193 AELogEntry* iter = newQueue;
194 while (cur) {
195 iter->copyEntry(*cur);
196 iter = iter->m_pNextNode;
197 cur = cur->m_pNextNode;
198 }
199 delete[] ptr;
200 ptr = newQueue;
201 }
202
203 return AELP_ERR_NOERROR;
204}

206AELogEntryInfo AELogParser::findNextEntry(const cint severity, const std::string_view mname, const bool strictSeverity) {
207
208 if (this->isClosed()) {
209 return AELogEntryInfo::invalidEntry();
210 }
211
212 std::string mnameStr = mname.data();
213
214 if (!mname.empty() && !this->m_mapModuleNames.contains(mnameStr)) {
215 return AELogEntryInfo{ .mnameIndex = AELEI_INVLAID_MNAME };
216 }
217
218 if (!ace::utils::isInRange(AELP_SEVERITY_ALL, AELOG_TYPE_FATAL_ERROR, severity) || //is in range at all?
219 !this->containsSeverity(severity, strictSeverity)) { // does this exist in the log (accounting for strict severity filter)
220 return AELogEntryInfo{ .logType = AELEI_INVALID_TYPE };
221 }
222
223
224 const short filterMnameIndex = (!mname.empty()) ? this->m_mapModuleNames[mnameStr].second : 0;
225
226 mnameStr.clear();
227 mnameStr.shrink_to_fit(); //not necessary anymore
228
229 const std::size_t entryNum = m_ullCurrentEntry; //write down the cursor value
230
231 //searching time
232 while (1) {
233
234 if (++this->m_ullCurrentEntry >= this->m_vecEntryIndices.size()) {
235 // check if cursor was already out of bounds
236 if ((entryNum + 1) >= this->m_vecEntryIndices.size()) {
237 return AELogEntryInfo{ .cursorIndex = AELEI_INVALID_CURSOR };
238 }
239 //nope, we got out of bounds ourselves
240 //because of invalid filter!
241
242 this->m_ullCurrentEntry = entryNum; // return the cursor to the previous position
243 return AELogEntryInfo {
244 //god damn it msvc. Why is it an error to do it *that* way, while initialising it with a number outside of ternary is okay
245 .mnameIndex = (short) ((filterMnameIndex) ? AELEI_INVLAID_MNAME : 0),
246 .logType = (cint) ((severity != AELP_SEVERITY_ALL) ? AELEI_INVALID_TYPE : 0)
247 };
248
249 }
250 if (AELogParser::checkSeverity(this->m_vecEntryIndices[this->m_ullCurrentEntry].logType, severity, strictSeverity) && //check the severity with lambdas
251 AELogParser::checkMName(this->m_vecEntryIndices[this->m_ullCurrentEntry].mnameIndex, filterMnameIndex)) {
252
253 break;
254 }
255 }
256
257 return this->m_vecEntryIndices[this->m_ullCurrentEntry];
