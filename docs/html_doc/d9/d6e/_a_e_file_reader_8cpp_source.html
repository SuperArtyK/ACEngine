20int AEFileReader::openFile(const std::string_view fname) {

53cint AEFileReader::readStringNL(char* str, const int dcount) noexcept {
54 _AEFR_EXIT_ON_READ_CLOSED_FILE;

78cint AEFileReader::readStringNULL(char* str, const llint dcount) {
79 _AEFR_EXIT_ON_READ_CLOSED_FILE;
80
81 if (dcount < 1 || !str) {
82 this->m_cLastError = AEFR_ERR_READ_ZERO_SIZE;
83 return AEFR_ERR_READ_ZERO_SIZE;
84 }
85
86 //clear buffer and read
87 std::memset(str, NULL, dcount + 1);
88 const cint temp = this->readData_ptr(str, dcount, sizeof(char));
89
90 //look for the null char and move the read cursor to the next char after null (if appeared)
91 const char* const nulchar = std::strchr(str, '\0');
92 fseek(m_fpFilestr, -llint((dcount - (nulchar - str) - 1)), SEEK_CUR);
93 this->m_szLastReadAmount = std::strlen(str);
94
95 return temp;
96}

98cint AEFileReader::readBoolString(bool& num) noexcept {
99 _AEFR_EXIT_ON_READ_CLOSED_FILE;
100 char str[6]{};
101
102 cint temp = this->readData_ptr(str, 4, sizeof(char)); //read possible "true"
103 for (int i = 0; i < 4; i++) {
104 str[i] = std::tolower(str[i]);
105 }
106
107 if (!std::memcmp(str, "true", 5)) {
108 num = true;
109 return temp;
110 }
111
112 temp = this->readData_ptr(&str[4], 1, sizeof(char)); //read last letter to make up "false"
113 if (!std::memcmp(str, "false", 6)) {
114 num = false;
115 return temp;
116 }
117 return AEFR_ERR_READING_ERROR; //whoops, nothing to read!
118}

121cint AEFileReader::readData_ptr(void* cdata, const std::size_t dcount, const std::size_t dsize) noexcept {
122 _AEFR_EXIT_ON_READ_CLOSED_FILE;
123
124 if (!dcount || !dsize || !cdata) {
125 this->m_cLastError = AEFR_ERR_READ_ZERO_SIZE;

127 }
128
129 this->m_szLastReadAmount = fread(cdata, dsize, dcount, this->m_fpFilestr);
130 this->m_ullTotalReads++;
131 if (this->m_szLastReadAmount != dcount) {
132 if (std::feof(this->m_fpFilestr)) {
133 return AEFR_ERR_READING_EOF;
134 }
135 if (std::ferror(this->m_fpFilestr)) {
136 return AEFR_ERR_READING_ERROR;
137 }
138 }
139 return AEFR_ERR_READ_SUCCESS;
