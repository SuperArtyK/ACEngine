32cint AEFileReader::openFile(const std::string_view fname) {
33
34 if (this->isOpen()) {
35 return AEFR_ERR_OPEN_FILE_ALREADY_OPENED;
36 }
37
38
39 if (fname.empty()) {
40 return AEFR_ERR_OPEN_FILE_NAME_EMPTY;
41 }
42
43 this->m_szLastReadAmount = 0;
44 this->m_fpFilestr = ace::utils::fopenCC(fname.data(), "rb");
45
46 if (this->isClosed()) {
47 // this can also be triggered if program has no permission to access the file
48 // but generally it just would be the absence of file
49 return AEFR_ERR_OPEN_FILE_DOESNT_EXIST;
50 }
51
52 this->m_sFilename = fname;
53 return AEFR_ERR_NOERROR;
54}

69//reads the string untill the newline or, as a safety measure, untill dcount. MUST be of the dcount+1 size!
70cint AEFileReader::readStringNL(char* str, const int dcount) noexcept {
71 _AEFR_EXIT_ON_READ_CLOSED_FILE;
72
73 if (!str || !dcount) {
74 return AEFR_ERR_READ_ZERO_SIZE;
75 }
76
77 std::memset(str, NULL, std::size_t(dcount) + 1);
78 const bool temp = std::fgets(str, dcount, this->m_fpFilestr);
79 this->m_szLastReadAmount = std::strlen(str);
80 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
81
82 if (!temp) {
83 return AEFR_ERR_READ_ERROR;
84 }
85
86 if (std::feof(this->m_fpFilestr)) {
87 return AEFR_ERR_READ_EOF;
88 }
89 return AEFR_ERR_READ_SUCCESS;

92//reads the string untill the null, or untill dcount
93//moves the cursor to the found null char+1
94cint AEFileReader::readStringNULL(char* str, const llint dcount) {
95 _AEFR_EXIT_ON_READ_CLOSED_FILE;
96
97 if (!str || dcount < 1) {
98 return AEFR_ERR_READ_ZERO_SIZE;
99 }
100
101 //clear buffer and read
102 std::memset(str, NULL, dcount + 1);
103 const cint temp = this->readData_ptr(str, dcount, sizeof(char));
104
105 //look for the null char and move the read cursor to the next char after null (if appeared)
106 const char* const nulchar = std::strchr(str, '\0');
107 fseek(m_fpFilestr, -llint((dcount - (nulchar - str) - 1)), SEEK_CUR);
108 this->m_szLastReadAmount = std::strlen(str);
109
110 return temp;
111}

113cint AEFileReader::readBoolString(bool& num) noexcept {
114 _AEFR_EXIT_ON_READ_CLOSED_FILE;
115 char str[sizeof("false")]{};
116
117 cint temp = this->readData_ptr(str, 4, sizeof(char)); //read possible "true"
118 for (int i = 0; i < 4; i++) {
119 str[i] = std::tolower(str[i]);
120 }
121
122 if (!std::memcmp(str, "true", sizeof("true"))) {
123 num = true;
124 return temp;
125 }
126
127 temp = this->readData_ptr(&str[4], 1, sizeof(char)); //read last letter to make up "false"
128 if (!std::memcmp(str, "false", sizeof("false"))) {
129 num = false;
130 return temp;
131 }
132 return AEFR_ERR_READ_ERROR; //whoops, nothing to read!
133}

136cint AEFileReader::readData_ptr(void* cdata, const std::size_t dcount, const std::size_t dsize) noexcept {
137 _AEFR_EXIT_ON_READ_CLOSED_FILE;
138
139 if (!cdata || !dcount || !dsize) {
140 return AEFR_ERR_READ_ZERO_SIZE;
141 }
142
143 this->m_szLastReadAmount = fread(cdata, dsize, dcount, this->m_fpFilestr);
144 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
145 if (this->m_szLastReadAmount != dcount) {
146 if (std::feof(this->m_fpFilestr)) {
147 return AEFR_ERR_READ_EOF;
148 }
149 if (std::ferror(this->m_fpFilestr)) {
150 return AEFR_ERR_READ_ERROR;
151 }
152 }
153 return AEFR_ERR_READ_SUCCESS;
154}
