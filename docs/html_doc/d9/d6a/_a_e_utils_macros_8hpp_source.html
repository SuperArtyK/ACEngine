42/// Macro for the calculation of the **size of the static array** (that is determined compile-time).
43/// <param name="arr">The array to calculate the size of</param>
44/// @warning If used on dynamic array, it yields the [size of the pointer on the platform] divided by [size of the array's type]

57/// Macro for the better-fomatted **assertion for runtime checks**.
58/// <param name="expression">The logical expression to assert</param>
59/// <param name="message">The message to display if the assertion fails</param>
60/// @note Only works in engine's debug mode. Does nothing if #ENGINE_DEBUG is not set to 1
61/// @note Only works with engine's assertions enabled. Does nothing if #ENGINE_ENABLE_ASSERT is not set to 1
62/// @see #ENGINE_DEBUG
63/// @see #ENGINE_ENABLE_ASSERT
64#if ENGINE_ENABLE_ASSERT && ENGINE_DEBUG
65#define aceAssert(expression, message) if ( !(expression) ) { \
66 std::cerr << "\n\nASSERTION FAILED! \nExpression: " << #expression << \
67 "\nFile: " << __FILE__ << \
68 "\nFunction: " << __FUNCTION__ << "()" << \
69 "\nLine: " << __LINE__ << \
70 "\nMessage: " << message << std::endl; \
71 std::terminate(); }
72#else
73#define aceAssert(expression, message) {}
74#endif

76/// Macro for the **debug printf()** printing, use the same way as you have used printf before.
77/// Prints "[DEBUG] [function name that called it] [invoked line] -> [stuff that you wanted the printf to print]"
78/// @note Prints additional newline with each print
79/// @note Only works in engine's debug mode. Does nothing if #ENGINE_DEBUG is not set to 1
80/// @see #ENGINE_DEBUG
81#if ENGINE_DEBUG
82#define dprintf(...) if constexpr (ENGINE_DEBUG) { \
83 std::printf("[DEBUG] [%s()] [line:%llu] -> ", \
84 __FUNCTION__, \
85 (ullint)__LINE__); \
86 std::printf(__VA_ARGS__); \
87 std::printf("\n"); }
88#else
89#define dprintf(...)
90#endif

92/// Macro to check **if the given type T is the same as Y**.
93/// <param name="T">The first type to compare</param>
94/// <param name="Y">The second type to compare</param>
95/// @note This discards the cv-qualifiers from the types in the comparison
96#define IS_SAME_NOCV(T, Y) (std::is_same<std::remove_cv<T>::type, std::remove_cv<Y>::type>::value)
