37

60int AEFileWriter::openFile(const std::string_view str, const cint flags, const ullint af_interval) {
61
62 if (this->m_fpFilestr) { // open already -> pls close
63 this->closeFile();
64 }
65
66 if (str.empty()) {
67 this->m_sFilename.clear();
68 this->m_cLastError = AEFW_ERR_FILE_NAME_EMPTY;
69 return AEFW_ERR_FILE_NAME_EMPTY;
70 }
71
72 this->m_szLastWrittenAmount = 0;
73 this->m_sFilename = str;
74 std::replace(m_sFilename.begin(), m_sFilename.end(), '\\', '/');
75 const std::size_t found = this->m_sFilename.rfind('/');
76 if (found != std::string::npos) {
77 std::filesystem::create_directories(this->m_sFilename.substr(0, found));
78 }
79
80 switch (flags) {
81
82 case AEFW_FLAG_APPEND://cursor at end, allow changing cursor position
83 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "r+b");
84 if (!this->m_fpFilestr) {//couldnt open, force create
85 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "wb");
86 if (!this->m_fpFilestr) {
87 break; // some other error
88 }
89 }
90 fseek(this->m_fpFilestr, 0L, SEEK_END);
91 break;
92
93 case AEFW_FLAG_APPEND_NO_CURSOR_MOVE://only appending, not allowing to change cursor position
94 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "ab");
95 break;
96
97
98 case AEFW_FLAG_NOFLAGS://no flags, defaulting to truncation
99 case AEFW_FLAG_TRUNCATE://truncate file
100 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "wb");
101 break;
102
103 default:
104 this->m_cLastError = AEFW_ERR_FILE_WRONG_FLAG; //wrong flag, not opening the file
105 return AEFW_ERR_FILE_WRONG_FLAG;
106 }
107
108 // last check, to see if everything is okay
109 if (!this->m_fpFilestr) {
110 this->m_cLastError = AEFW_ERR_FILE_OPEN_ELSE; //file is still somehow nonexistent
111 return AEFW_ERR_FILE_OPEN_ELSE;
112 }
113
114 //then everything is good,
115 this->m_cFlags = flags;
116 this->m_ullFlushInterval = af_interval;
117 return AEFW_ERR_NOERROR;
118}
