35cint AEFileWriter::writeData_ptr(const void* const cdata, const std::size_t dcount, const std::size_t dsize, const bool useAutoFlush) noexcept {
36
37 _AEFW_EXIT_ON_WRITE_CLOSED_FILE;
38
39 if (!cdata || !dcount || !dsize) {
40 return AEFW_ERR_WRITE_ZERO_SIZE;
41 }
42
43 this->m_szLastWrittenAmount = fwrite(cdata, dsize, dcount, this->m_fpFilestr);
44 this->m_ullTotalWrites.fetch_add(1, std::memory_order::relaxed);
45 if (useAutoFlush) {
46 this->autoFlush();
47 }
48
49 if (m_szLastWrittenAmount != dcount) {
50 return AEFW_ERR_WRITE_ERROR;
51 }
52
53 return AEFW_ERR_WRITE_SUCCESS;
54}

57// open file with flags
58cint AEFileWriter::openFile(const std::string_view str, const cint flags, const ullint af_interval) {
59
60 if (this->isOpen()) { // open already -> pls close
61 return AEFW_ERR_OPEN_FILE_ALREADY_OPENED;
62 }
63
64 if (str.empty()) {
65 this->m_sFilename.clear();
66 return AEFW_ERR_OPEN_FILE_NAME_EMPTY;
67 }
68
69 this->m_szLastWrittenAmount = 0;
70 this->m_sFilename = str;
71 std::replace(m_sFilename.begin(), m_sFilename.end(), '\\', '/');
72 const std::size_t found = this->m_sFilename.rfind('/');
73 if (found != std::string::npos) {
74 std::filesystem::create_directories(this->m_sFilename.substr(0, found));
75 }
76
77 switch (flags) {
78
79 case AEFW_FLAG_APPEND://cursor at end, allow changing cursor position
80 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "r+b");
81 if (this->isClosed()) {//couldn't open, force create
82 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "wb");
83 if (this->isClosed()) {
84 break; // some other error
85 }
86 }
87 fseek(this->m_fpFilestr, 0L, SEEK_END);
88 break;
89
90 case AEFW_FLAG_APPEND_NO_CURSOR_MOVE://only appending, not allowing to change cursor position
91 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "ab");
92 break;
93
94
95 case AEFW_FLAG_NOFLAGS://no flags, defaulting to truncation
96 case AEFW_FLAG_TRUNCATE://truncate file
97 this->m_fpFilestr = ace::utils::fopenCC(str.data(), "wb");
98 break;
99
100 default:
101 return AEFW_ERR_OPEN_FILE_WRONG_FLAG;
102 }
103
104 // last check, to see if everything is okay
105 if (this->isClosed()) {
106 return AEFW_ERR_OPEN_FILE_ELSE;
107 }
108
109 //then everything is good,
110 this->m_cFlags = flags;
111 if (this->m_ullFlushInterval != AEFW_DEFAULT_AUTOFLUSH_INTERVAL) {
112 this->m_ullFlushInterval = af_interval;
113 }
114 return AEFW_ERR_NOERROR;
115}
