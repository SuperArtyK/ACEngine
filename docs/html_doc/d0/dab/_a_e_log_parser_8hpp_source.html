382 constexpr bool errorFromAELEI(const AELogEntryInfo& leInfo) const noexcept {
383 if (leInfo.isInvalid()) {
384 return AEFR_ERR_FILE_NOT_OPEN;
385 }
386 if (leInfo.isBroken()) {
387 if (leInfo.isInvalidCursor()) {
388 return AEFR_ERR_READ_EOF;
389 }
390 if (leInfo.isInvalidMName()) {
391 return AELP_ERR_INVALID_MODULE_NAME;
392 }
393 if (leInfo.isInvalidType()) {
394 return AELP_ERR_INVALID_SEVERITY;
395 }
396 }
397 // everything is okay
398 return AELP_ERR_NOERROR;
399 }
