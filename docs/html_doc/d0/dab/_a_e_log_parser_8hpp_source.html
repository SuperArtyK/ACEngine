105 cint parseEntryString(AELogEntry& entry, const std::string_view entryString) {
106 constexpr std::size_t POS_TYPE = 22, POS_MNAME = 39;
107 if (entryString.size() == AELOG_ENTRY_MAX_SIZE + 1 || entryString.size() < 47) { // it is more then the max size or less than possible size -- invalid or...somehow modified
108 return AELP_ERR_INVALID_LENGTH;
109 }
110
111 char logmessage[512]{}; //log message, exactly 511 characters
112 char mname[33]{}; // +1 more character to see if it's not correct format
113 char ltype[15]{}; //14 characters guaranteed
114
115 //check the log entry's time
116 const std::time_t entryTime = ace::utils::stringToDate(entryString.data(), "[%Y-%m-%d.%X");
117 if (entryTime == -1) {
118 return AELP_ERR_INVALID_TIME;
119 }
120
121 // check for the "] " between the time and type
122 if (std::memcmp(entryString.data() + POS_TYPE - 2, "] ", 2)) {
123 return AELP_ERR_INVALID_FORMAT;
124 }
125
126 // read the possible type
127 if (sscanf(entryString.data() + POS_TYPE, "[%14[^]] ", ltype) != 1) {
128 return AELP_ERR_INVALID_TYPE; //oops!
129 }
130
131 // check for the valid log entry's type
132 const cint entryType = AELogParser::matchLogType(ltype);
133 if (entryType == -1) {
134 return AELP_ERR_INVALID_TYPE; //oops!
135 }
136
137
138 // check for the "] " between the type and module name
139 if (std::memcmp(entryString.data() + POS_MNAME - 2, "] ", 2)) {
140 return AELP_ERR_INVALID_FORMAT;
141 }
142
143 // read the possible module name
144 if (sscanf(entryString.data() + POS_MNAME, "[%32[^]] ", mname) != 1) {
145 return AELP_ERR_INVALID_MNAME;
146 }
147
148 // check for the correct module name
149 const std::string_view strvMname(mname);
150 if (strvMname.size() > 31 || !ace::utils::isAlNumUs(strvMname)) {
151 return AELP_ERR_INVALID_MNAME;
152 }
153
154 //okay. we might have read that.
155 //Since the module name is variable and cannot be predicted
156 //We might have read untill the end of the "entry"
157 //so check it!
158 if (entryString.size() < (POS_MNAME + strvMname.size() + 5)) { //position of module name (39)+ size of the mname + 5 characters forward (leading [, trailing "]: ", and 1 character for message)
159 return AELP_ERR_INVALID_LENGTH;
160 }
161
162 //cool, now onwards to text
163 //check for the existing boundary between module name and message
164 if (std::memcmp(entryString.data() + POS_MNAME + strvMname.size() + 1, "]: ", 3)) {
165 return AELP_ERR_INVALID_FORMAT; //oops
166 }
167
168 //cool, passed. now read untill the end
169 //the newline and or the size is guaranteed to exist
170 sscanf(entryString.data() + POS_MNAME + strvMname.size() + 4, "%[^511\n]", logmessage);
171
172 //time to write stuff
173 AELogEntry::clearEntry(&entry);
174
175 // what if the entry is a debug one?
176 if (entryType == AELOG_TYPE_DEBUG && std::strstr(logmessage, "DEBUG->") != nullptr) {
177 std::memcpy(entry.m_sLogMessage, logmessage + sizeof("DEBUG->") - 1, AELOG_ENTRY_MESSAGE_SIZE - sizeof("DEBUG->") + 1);
178 }
179 else {
180 std::memcpy(entry.m_sLogMessage, logmessage, AELOG_ENTRY_MESSAGE_SIZE);
181 }
182
183 std::memcpy(entry.m_sModuleName, mname, strvMname.size()); //module name
184 entry.m_tmLogTime = entryTime; //time
185 // not doing the queue..because it's the user's concern
186 entry.m_cStatus = AELOG_ENTRY_STATUS_READY; //status
187 entry.m_cLogType = entryType; //type
188
189 return AELP_ERR_NOERROR;
190 }
