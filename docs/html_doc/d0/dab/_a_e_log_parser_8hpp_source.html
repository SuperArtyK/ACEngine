84 cint fileRead = AEFR_ERR_NOERROR;
85 cint parsedEntry = AELP_ERR_NOERROR;
86 ullint cursor = 0;
87 AELogEntry dummyentry{};
88
89 while (1) {

91 fileRead = this->m_frLogReader.readStringNL(fileline, sizeof(fileline) - 1);
92
93 if (fileline[0] == '\n') { //first character is an empty line -- skipping...
94 break;
95 }
96 parsedEntry = AELogEntry::parseStringEntry(dummyentry, fileline, AELE_PARSE_STRING_WRITE_TYPE);
97
98 if (parsedEntry != AELP_ERR_NOERROR) {
99 if (fileRead != AEFR_ERR_NOERROR) {
100 break;
101 }
102 this->m_vecInvalidEntryIndices.push_back(cursor);
103 this->m_arrEntryAmount[0]++;
104 }
105 else {
106 this->m_vecEntryIndices.emplace_back(cursor, dummyentry.m_cLogType);
107 this->m_arrEntryAmount[dummyentry.m_cLogType + 1]++;
108 }
109 }

112
