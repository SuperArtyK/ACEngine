24//Error flags
25/// Macro for the indicator that everything is good/no error was encountered in the process
26#define AELP_ERR_NOERROR ENGINE_MODULE_ERR_NOERROR
27/// Macro for the error when the passed module name doesn't match the parsed modulenames in the log
28/// @see AELogParser::findNextEntry()
29#define AELP_ERR_INVALID_MODULE_NAME -20
30/// Macro for the error when the passed severity is invalid in the current context
31/// Ex. non-existent severity, or the severity wasn't found in the log file
32#define AELP_ERR_INVALID_SEVERITY -21
33/// Macro for the error when the passed severity and modulename filter didn't produce a result when applied
34/// Ex. both the severity and the module name exist in the parsed log file, but never together in the same entry
35/// @see AELogParser::findNextEntry()
36#define AELP_ERR_FILTER_NO_MATCHES -22
37/// Macro for the error when the passed severity and module name filter is invalid (e.g. empty)
38#define AELP_ERR_INVALID_FILTER -23
39/// Macro for the error when the null queue pointer was passed into the functions
40#define AELP_ERR_INVALID_QUEUE -24
41/// Macro for the severity value that includes all entries/severities in parsing
42/// Essentially disables severity filtering
43/// @note It disables (makes useless) the "strict severity" in the functions that set it
44/// @see AELogParser::findNextEntry()
45#define AELP_SEVERITY_ALL AELOG_TYPE_INVALID
46/// Macro for the "no"/empty modulename to pass for parsing

54/// <summary>
55/// ArtyK's Engine's Log Parser; it parses the log files that AELogger writes.
56/// Wrapper around AELogEntry for parsing and AEFileWriter for reading functionality, (ab)using them both.
57///
58/// When opening a file, it reads it and indexes it for log information,
59/// and after the parsing/reading the log file to memory can be requested (one entry at a time/the whole thing).
60/// Such info would be the amount of entries, amount of entries of each type, etc.
61/// Also it allows to filter the log by severity and read only important (to the use-case) data.
62///
63/// Hungarian notation is lp. (m_lpMyLogParser)
64/// Flags start with AELP_
65/// </summary>

157 /// <summary>
158 /// Read the next indexed *valid* log entry in the file of the given severity and module name filter, and return it's file cursors in the file
159 /// @note The severity value just changes the lowest limit of the log severity (lowest by default is debug). If a higher severity is encountered, it's read as well.
160 /// @note AELOG_TYPE_INVALID works the same as AELOG_TYPE_DEBUG. This function parses only *valid* entries.
161 /// </summary>
162 /// <param name="severity">The lowest severity of the log to find</param>
163 /// <param name="strictSeverity">The flag to indicate whether the search for severity should be strict (exact)</param>
164 /// <param name="mname">The module name of the log entry to search for</param>
165 /// <returns>The file cursors of the next valid entry (in the currently-opened log file); error codes from AELogParser::errorFromAELEI()</returns>
166 /// @see AELogParser::errorFromAELEI()
167 inline llint nextEntryCursor(const cint severity = AELOG_TYPE_DEBUG, const bool strictSeverity = false, const std::string_view mname = AELP_NO_MODULENAME) {
168
169 const AELogEntryInfo leInfo = this->findNextEntry(severity, mname, strictSeverity);
170 const cint ret = AELogParser::errorFromAELEI(leInfo);
171 if (ret != AELP_ERR_NOERROR) {
172 return ret; // return the error code from the AELEI
173 }
174
175 return leInfo.cursorIndex;
176 }

332 /// <summary>
333 /// Get the amount of valid entries in the log (with optional lowest severity setting).
334 /// Valid entries are entries that are not AELOG_TYPE_INVALID
335 /// </summary>
336 /// <param name="severity">The lowest severity of the log to find</param>
337 /// <returns>ULLINT_MAX if the severity is outside of the AELOG_TYPE_* range; ullint amount of entries</returns>
338 inline ullint amountEntriesValid(const cint severity = AELOG_TYPE_DEBUG) const noexcept {
339 if (!ace::utils::isInRange(AELOG_TYPE_INVALID, AELOG_TYPE_FATAL_ERROR, severity)) {
340 return ULLINT_MAX;
341 }
342 ullint tempres = 0;
343 for (int i = severity; i <= AELOG_TYPE_FATAL_ERROR; i++) {
344 tempres += this->m_arrEntryAmount[i + 1];
345 }
346 return tempres;
347 }

382 /// <summary>
383 /// Checks if the current log file contains the given severity (or higher, depending on the "strict-severity" flag)
384 /// </summary>
385 /// <param name="severity">The severity to search for</param>
386 /// <param name="strictSeverity">The flag to indicate whether the search for severity should be strict (exact)</param>
387 /// <returns>True if the log contains the severity, false otherwise; false if the invalid severity was passed or file is closed</returns>
388 inline bool containsSeverity(const cint severity, const bool strictSeverity = false) const noexcept {
389 if (!ace::utils::isInRange(AELP_SEVERITY_ALL, AELOG_TYPE_FATAL_ERROR, severity) || this->isClosed()) {
390 return false;
391 }
392 if (strictSeverity) {
393 return this->m_arrEntryAmount[severity + 1];
394 }
395 for (int i = ((severity == AELP_SEVERITY_ALL) ? AELOG_TYPE_DEBUG : severity); i <= AELOG_TYPE_FATAL_ERROR; i++) {
396 if (this->m_arrEntryAmount[i + 1]) {
397 return true;
398 }
399 }
400 return false;
401 }

478 /// <summary>
479 /// Determines the error codes from the invalid values in the passed AELogEntryInfo instance
480 /// </summary>
481 /// <param name="leInfo">The passed AELogEntryInfo instance to check</param>
482 /// <returns>AELP_ERR_NOERROR if everything is okay; AEFR_ERR_FILE_NOT_OPEN if it's fully invalid; AEFR_ERR_READ_EOF on invalid cursor; AELP_ERR_INVALID_MODULE_NAME on invalid module name; AELP_ERR_INVALID_SEVERITY on invalid type</returns>
483 static constexpr cint errorFromAELEI(const AELogEntryInfo& leInfo) noexcept {
484 if (leInfo.isInvalid()) {
485 return AEFR_ERR_FILE_NOT_OPEN;
486 }
487 if (leInfo.isBroken()) {
488 if (leInfo.isInvalidMName() && leInfo.isInvalidType()) {
489 return AELP_ERR_FILTER_NO_MATCHES;
490 }
491 if (leInfo.isInvalidCursor()) {
492 return AEFR_ERR_READ_EOF;
493 }
494 if (leInfo.isInvalidMName()) {
495 return AELP_ERR_INVALID_MODULE_NAME;
496 }
497 if (leInfo.isInvalidType()) {
498 return AELP_ERR_INVALID_SEVERITY;
499 }
500 }
501 // everything is okay
502 return AELP_ERR_NOERROR;
503 }

505 /// <summary>
506 /// Checks if the passed severity is valid compared to the filtering severity.
507 /// Moved from a lambda, so it's easier to use with other parts of the log parser.
508 /// @note If strictSeverity is true, performs exact check. Otherwise if the passed entrySeverity is more or equal to the filtering filterSeverity.
509 /// @note If filterSeverity is AELP_SEVERITY_ALL, the function always returns true
510 /// </summary>
511 /// <typeparam name="strictSeverity">Whether to check if both severity levels *match exactly*. Default: false</typeparam>
512 /// <param name="entrySeverity">The severity to check/filter</param>
513 /// <param name="filterSeverity">The filtering severity</param>
514 /// <param name="strictSeverity">The flag to indicate whether the severity check should be strict (exact)</param>
515 /// <returns>True if the filtering conditions are satisfied; false otherwise</returns>
516 static constexpr bool checkSeverity(const cint entrySeverity, const cint filterSeverity, const bool strictSeverity = false) noexcept {
517 if (filterSeverity == AELP_SEVERITY_ALL) {
518 return true;
519 }
520 if (strictSeverity) {
521 return entrySeverity == filterSeverity;
522 }
523 return entrySeverity >= filterSeverity;
524 }

526 /// <summary>
527 /// Checks if the passed entry's and filter's module name indices match
528 /// Moved from a lambda, so it's easier to use with other parts of the log parser.
529 /// @note If filterNameIndex is 0, then it will always return true (we aren't checking for it)
530 /// </summary>
531 /// <param name="entryNameIndex">The module name index of the entry</param>
532 /// <param name="filterNameIndex">The module name index filter</param>
533 /// <returns></returns>
534 static constexpr bool checkMName(const short entryNameIndex, const short filterNameIndex) noexcept {
535 if (!filterNameIndex) {
536 return true; // if the module name is empty -- return true always (we aren't checking for it)
537 }
538 return entryNameIndex == filterNameIndex;
539 }
