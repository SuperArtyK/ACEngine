174 inline llint nextEntryCursor(const cint severity = AELOG_TYPE_DEBUG) {
175 _AELP_CHECK_IF_FILE_OPEN;
176
177 //cycle the current entry number untill we find an entry with proper severity level
178 while (1) {
179 if (++this->m_ullCurrentEntry >= this->m_vecEntryIndices.size()) {
180 return AEFR_ERR_READ_EOF;
181 }
182 if (this->m_vecEntryIndices[this->m_ullCurrentEntry].second >= severity) {
183 break;
184 }
185 }
186 return this->m_vecEntryIndices[this->m_ullCurrentEntry].first;
187 }

302 inline ullint amountValidEntries(const cint severity = AELOG_TYPE_DEBUG) const noexcept {
303 if (!ace::utils::isInRange<cint>(AELOG_TYPE_INVALID, AELOG_TYPE_FATAL_ERROR, severity)) {
304 return ULLINT_MAX;

306 ullint tempres = 0;
