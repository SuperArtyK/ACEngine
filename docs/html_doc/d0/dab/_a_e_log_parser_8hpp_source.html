76 char fileline[AELE_FORMAT_MAX_SIZE + 1]{}; //+1 more character to determine the validity)
77 cint fileRead = AEFR_ERR_NOERROR;
78 cint parsedEntry = AELP_ERR_NOERROR;
79 ullint cursor = 0;
80 AELogEntry dummyentry{};
81
82 while (1) {
83 cursor = this->m_frLogReader.getCursorPos();
84 fileRead = this->m_frLogReader.readStringNL(fileline, sizeof(fileline) - 1);
85
86 if (fileline[0] == '\n') { //first character is an empty line -- skipping...
87 break;
88 }
89 parsedEntry = AELogEntry::parseStringEntry(dummyentry, fileline, AELE_PARSE_STRING_WRITE_TYPE);
90
91 if (parsedEntry != AELP_ERR_NOERROR) {
92 if (fileRead != AEFR_ERR_NOERROR) {
93 break;
94 }
95 this->m_vecInvalidEntryIndices.push_back(cursor);
96 this->m_arrEntryAmount[0]++;
97 }
98 else {
99 this->m_vecEntryIndices.emplace_back(cursor, dummyentry.m_cLogType);
100 this->m_arrEntryAmount[dummyentry.m_cLogType + 1]++;
101 }
102 }
