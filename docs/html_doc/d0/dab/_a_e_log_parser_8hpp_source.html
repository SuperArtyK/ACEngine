477 static constexpr cint errorFromAELEI(const AELogEntryInfo& leInfo) noexcept {
478 if (leInfo.isInvalid()) {
479 return AEFR_ERR_FILE_NOT_OPEN;
480 }
481 if (leInfo.isBroken()) {
482 if (leInfo.isInvalidMName() && leInfo.isInvalidType()) {
483 return AELP_ERR_FILTER_NO_MATCHES;
484 }
485 if (leInfo.isInvalidCursor()) {
486 return AEFR_ERR_READ_EOF;
487 }
488 if (leInfo.isInvalidMName()) {
489 return AELP_ERR_INVALID_MODULE_NAME;
490 }
491 if (leInfo.isInvalidType()) {
492 return AELP_ERR_INVALID_SEVERITY;
493 }
494 }
495 // everything is okay
496 return AELP_ERR_NOERROR;
497 }

510 static constexpr bool checkSeverity(const cint entrySeverity, const cint filterSeverity, const bool strictSeverity = false) noexcept {
511 if (filterSeverity == AELP_SEVERITY_ALL) {
512 return true;
513 }
514 if (strictSeverity) {
515 return entrySeverity == filterSeverity;
516 }
517 return entrySeverity >= filterSeverity;
518 }
