361 template<typename T, typename Y = T>
362 constexpr bool equals(const T num, const Y num2) noexcept {
363
364
365 if constexpr (std::is_floating_point<T>::value && !std::is_floating_point<Y>::value) { // 1st type is float, 2nd one isn't
366 return ace::math::fequals(num, T(num2));
367 }
368 if constexpr (!std::is_floating_point<T>::value && std::is_floating_point<Y>::value) { // 1st type isn't float, 2nd one is
369 return ace::math::fequals(Y(num), num2);
370 }
371 if constexpr (std::is_floating_point<T>::value && std::is_floating_point<Y>::value) { // both are float
372 if constexpr (std::numeric_limits<T>::max_digits10 > std::numeric_limits<Y>::max_digits10) { // 1st type is larger
373 return ace::math::fequals(num, T(num2));
374 }
375 return ace::math::fequals(Y(num), num2); // 2nd type is larger
376 }
377 else {
378 return (num == num2);
379 }

389 constexpr T sqrt(const T num) noexcept {
390 if (num < 0) {
391 return T(-1);
392 }
393
394 T val[2] = { num,0 };
395 while (!ace::math::equals(val[0], val[1])) {
396 val[1] = val[0];
397 val[0] = (val[0] + num / val[0]) / T(2);
398 if constexpr (std::is_integral<T>::value) {
399 if (val[1] < val[0]) {
400 break;
401 }
402 }
403 }
404 return val[1];

414 template<typename T = long double>
415 constexpr T intPow(const T num, llint power) noexcept {
416
417 //switch to save up on conditionals and jumps
418 //yeah microoptimisation, but it's fine, doesn't hurt
419
420 switch (power) {
421
422 case -1:
423 if(ace::math::equals(num, T(0))) { return std::numeric_limits<T>::infinity(); } // 1/0 -- infinity to comply with floats
424 return T(1) / num;
425
426 case 0: // power is 0, anything to power 0 is 1
427 return 1;
428 break;
429
430 case 1: // power is 0, anything to power 1 is itself
431 return num;
432 break;
433
434 default:
435 if(ace::math::equals(num, T(1))) { return T(1); } // 1 to power of anything is still 1
436 break;
437 }
438
439 T res;
440
441 if (power > 1) {
442 res = num;
443 for (int i = 1; i < power; i++) {
444 res *= num;
445 }
446 }
447 else {
448 res = 1;
449 power = -power;
450 for (int i = 1; i < power; i++) {
451 res /= num;
452 }
453 }
454
455 return res;
456 }
