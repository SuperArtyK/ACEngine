349 template<typename T, typename Y>
350 constexpr bool equals(const T num, const Y num2) noexcept {
351
352
353 if constexpr (std::is_floating_point<T>::value && !std::is_floating_point<Y>::value) { // 1st type is float, 2nd one isn't
354 return ace::math::fequals(num, T(num2));
355 }
356 if constexpr (!std::is_floating_point<T>::value && std::is_floating_point<Y>::value) { // 1st type isn't float, 2nd one is
357 return ace::math::fequals(Y(num), num2);
358 }
359 if constexpr (std::is_floating_point<T>::value && std::is_floating_point<Y>::value) { // both are float
360 if constexpr (std::numeric_limits<T>::max_digits10 > std::numeric_limits<Y>::max_digits10) { // 1st type is larger
361 return ace::math::fequals(num, T(num2));
362 }
363 return ace::math::fequals(Y(num), num2); // 2nd type is larger
364 }
365 else {
366 return (num == num2);
367 }

376 template<typename T = long double>
377 constexpr T sqrt(const T num) noexcept {
378 if (num < 0) {
379 return T(-1);
380 }
381
382 T val[2] = { num,0 };
383 while (!ace::math::equals(val[0], val[1])) {
384 val[1] = val[0];
385 val[0] = (val[0] + num / val[0]) / T(2);
386 if constexpr (std::is_integral<T>::value) {
387 if (val[1] < val[0]) {
388 break;
389 }
390 }
391 }
392 return val[1];

402 template<typename T = long double>
403 constexpr T intPow(const T num, llint power) {
404
405 //switch to save up on conditionals and jumps
406 //yeah microoptimisation, but it's fine, doesn't hurt
407
408 switch (power) {
409
410 case -1:
411 if(ace::math::equals(num, T(0))) { return std::numeric_limits<T>::infinity(); } // 1/0 -- infinity to comply with floats
412 return T(1) / num;
413
414 case 0: // power is 0, anything to power 0 is 1
415 return 1;
416 break;
417
418 case 1: // power is 0, anything to power 1 is itself
419 return num;
420 break;
421
422 default:
423 if(ace::math::equals(num, T(1))) { return T(1); } // 1 to power of anything is still 1
424 break;
425 }
426
427 T res;
428
429 if (power > 1) {
430 res = num;
431 for (int i = 1; i < power; i++) {
432 res *= num;
433 }
434 }
435 else {
436 res = 1;
437 power = -power;
438 for (int i = 1; i < power; i++) {
439 res /= num;
440 }
441 }
442
443 return res;
444 }
