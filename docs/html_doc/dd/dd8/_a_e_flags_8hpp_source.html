28//Engine flags
29/// Macro to enable the **debug-marked features** of the engine.
30/// It's an engine thing, ***separate*** from debug/release compiler builds.
31/// @note Set to 1 to enable, and 0 to disable this feature
32#define ENGINE_DEBUG 1
33/// Macro for the engine's **default delay** in frames/ticks per second (used in AEFrame and AETimer).
34/// @note Set to any positive value to enable, and any negative value or 0 to disable this feature
35/// @attention Disabling this disables the sleep functionality in instances of AEFrame and AETimer that use the default delay.
36/// @see AEFrame
37/// @see AETimer
38#define ENGINE_FPS 60.0
39/// Macro to enable the engine's **global modules** (like AETimer, AELogger, etc).
40/// @note Set to 1 to enable, and 0 to disable this feature
41/// @remark Disabling this code *may* break your code (from compiling), as it removes the global module declarations completely
42#define ENGINE_ENABLE_GLOBAL_MODULES 1
43/// Macro to enable the engine's **assert code**
44/// @note This only works if the debug features are enabled (#ENGINE_DEBUG).
45/// @note Set to 1 to enable, and 0 to disable this feature
46/// @see #aceAssert()
47#define ENGINE_ENABLE_ASSERT 1
48/// Macro to enable the engine's **trigonometry lookup tables and code**.
49/// @note Set to 1 to enable, and 0 to disable this feature
50/// @see AETrigLookup.hpp
51#define ENGINE_ENABLE_MATH_LOOKUP 1

53//AEFileWriter flags
54/// Macro to enable the AEFileWriter's **auto-flushing mechanism**.
55/// @note Set to 1 to enable, and 0 to disable this feature
56/// @see AEFileWriter::autoFlush()
57#define AEFW_AUTOFLUSH_ENABLE 1
58/// Macro for the AEFileWriter's **default auto-flushing action** (default argument in file-writing functions).
59/// @note This only works/matters if the auto-flushing feature is enabled (#AEFW_AUTOFLUSH_ENABLE)
60/// @see AEFileWriter::autoFlush()
61#define AEFW_DEFAULT_AUTOFLUSH true
62/// Macro for the AEFileWriter's **default value to trigger auto-flushing**.
63/// @remark Default value (ullint)-1 means **never**. Or almost never, once about every 2^64 write operations.
64/// @note This only works/matters if the auto-flushing feature is enabled (#AEFW_AUTOFLUSH_ENABLE)
65/// @see AEFileWriter::autoFlush()
66#define AEFW_DEFAULT_AUTOFLUSH_INTERVAL (ullint)-1

68//AELogger flags
69/// Macro for the AELogger's **default queue size**
70/// @warning Setting this to a very large or a negative number can crash the engine on AELogger's creation.
71/// @warning All because the queue too is large to be allocated.
72#define AELOG_DEFAULT_QUEUE_SIZE (ullint)2048
73/// Macro for the AELogger's **default module/log invoker name**
74/// @warning This must follow the rules for the module name: Be alphanumeric+underscore and be no more than #AELE_MODULENAME_SIZE.
75/// @warning Otherwise any logs invoked with a default module name will fail.
76/// @see AELogger::writeToLog()
77/// @see AELogger::writeToLogDebug()
78/// @see AELogger::writeToLogDirectly()
79#define AELOG_DEFAULT_MODULE_NAME "ACEngine"
80/// Macro for the AELogger's **default reserve amount of the allocation vector** (which stores all queue pointers).
81/// @attention I personally *wouldn't* advise to play with it (but you can).
82/// @remark 32 allocations, 1.5 queue size factor -- 884 million entries. If each node is 568 bytes -- it all will eat up 468 gb of ram.
83/// @remark I think that's enough :)
84#define AELOG_DEFAULT_ALLOC_VECTOR_RESERVE 32
85/// Macro for the AELogger's **default log file path**
86/// @see AELogger::openLog()
87#define AELOG_DEFAULT_LOG_PATH "./logs/"

91/// @note Changing this -- changes the size of the AELogEntry structure

100/// * Too little will make the messages cut off
101/// * Too large will increase the memory usage by a lot
102/// @see AELogEntry::m_sModuleName[]

106/// A macro for the AEVector::operator[] to **wrap dimension index around** if the index is bigger than the dimension amount.
107/// Example: 2d vector a, possible values to the [] is 0 and 1. If you input 2, this will be equivalent of [0]; 3 -> [1], and so on.
108/// @note Set to 1 to enable, and 0 to disable this feature
