38/// A macro to **complete the module declaration with __AEModuleBase**.
39/// Under the hood, it finishes specialising __AEModuleBase::getModuleName() for the inheriting class.
40/// <typeparam name="T">Inheriting module's class name</typeparam>
41/// @note Usage: Always put it as the **last** entry to the class, before the closing brace **}**
42/// @warning If you won't do it and use the __AEModuleBase<T>::getModuleName(), the project won't compile with the "unresolved external" error
43/// @see __AEModuleBase
44#define REGISTER_MODULE(T) public: \
45 [[nodiscard]] static constexpr std::string_view getModuleName(void) noexcept { \
46 static_assert(sizeof(STR(T)) <= AELE_MODULENAME_SIZE, "The module name is TOO BIG! (check AELE_MODULENAME_SIZE in AEMisc/include/AEFlags.hpp"); \
47 return #T; \
48 }

66/// <summary>
67/// This is a **base class for all the modules in the engine**; all basic modules inherit from it.
68///
69/// This allows tracking the name of the module, the amount of alive instances and soon: name and number of the instance.
70///
71/// @remark Intended to be inherited from in class declaration, not direct usage.
72/// @note Usage: <br>
73/// class **[classname]** : public __AEModuleBase<**[classname]**>
74/// @warning If you use the __AEModuleBase<T>::getModuleName() and do not include #REGISTER_MODULE(T) after the whole class definition (before closing '}'), the program won't compile
75/// @see #REGISTER_MODULE(T)
76/// </summary>
77/// <typeparam name="T">The class name of the module to inherit from this class</typeparam>
78template<typename T>
79class __AEModuleBase
80{
81public:
82
83 /// <summary>
84 /// **Basic constructor** that increases module count.d
85 /// </summary>
86 __AEModuleBase(void) noexcept : m_ullModuleId(++m_ullInstanceCounter) {
87 this->m_ullModulesAlive++;
88 }
89
90 /// <summary>
91 /// **Basic copy constructor** that increases module count.
92 ///
93 /// @remark It doesn't get called, unless expicitly called
94 /// @note Doesn't do anything other than increase module count
95 /// </summary>
96 /// <param name="val">The object passed to the copy constructor (we don't do anything with it)</param>
97 __AEModuleBase(const __AEModuleBase<T>& val) noexcept : m_ullModuleId(++m_ullInstanceCounter) {
98 this->m_ullModulesAlive++;
99 }
100
101 /// <summary>
102 /// **Basic destructor** that decreases module count.
103 /// </summary>
104 virtual ~__AEModuleBase(void) noexcept {
105 this->m_ullModulesAlive--;
106 }
107
108 /// <summary>
109 /// **Basic assignment operator -- it does nothing, so it's deleted** (make your own).
110 /// </summary>
111 __AEModuleBase<T>& operator=(const __AEModuleBase<T>& val) = delete;
112
113 /// <summary>
114 /// Get the **amount of the currently-alive instances of module T**.
115 ///
116 /// @see __AEModuleBase<T>::m_ullModulesAlive
117 /// </summary>
118 /// <returns>
119 /// Unsigned long long of the alive module amount
120 /// </returns>
121 [[nodiscard]] static inline ullint getModuleAmount(void) noexcept {
122 return __AEModuleBase<T>::m_ullModulesAlive.load();
123 }
124
125 /// <summary>
126 /// Get the **name of the module**.
127 ///
128 /// @attention You *need* to add #REGISTER_MODULE() to the end of the class declarations if you want to use this thing
129 /// @see #REGISTER_MODULE()
130 /// </summary>
131 /// <returns>
132 /// The name of the module as a const std::strinv_view type
133 /// </returns>
134 [[nodiscard]] static constexpr const std::string_view getModuleName(void) noexcept;

136 /// <summary>

138 ///
139 /// @remark Every module has its own instance counter
140 /// </summary>
141 /// <returns>
142 /// The id number of the instance of the module as **ullint** type
143 /// </returns>
144 [[nodiscard]] constexpr const ullint getModuleID(void) noexcept {
145 return m_ullModuleId;
146 }
147
148 /// <summary>
149 /// Get the current value of the **instance counter** of the module.
150 ///
151 /// @see __AEModuleBase::m_ullInstanceCounter

153 /// <returns>
154 /// The amount of the module's instances overall created throughout the program's runtime
155 /// </returns>

157 return m_ullInstanceCounter;

159
160protected:
