65/// <summary>
66/// This is a **base class for all the modules in the engine**; all basic modules inherit from it.
67/// This allows tracking the name of the module, the amount of alive instances and soon: name and number of the instance.
68/// @remark Intended to be inherited from in class declaration, not direct usage.
69/// @note Usage: <br>
70/// class **[classname]** : public __AEModuleBase<**[classname]**>
71/// @warning If you use the __AEModuleBase<T>::getModuleName() and do not include #REGISTER_MODULE(T) after the whole class definition (before closing '}'), the program won't compile
72/// @see #REGISTER_MODULE(T)
73/// </summary>
74/// <typeparam name="T">The class name of the module to inherit from this class</typeparam>
75template<typename T>
76class __AEModuleBase
77{
78public:
79
80 /// <summary>
81 /// **Basic constructor** that increases module count
82 /// </summary>
83 __AEModuleBase<T>(void) noexcept {
84 this->m_ullModulesAlive++;
85 }
86
87 /// <summary>
88 /// **Basic copy constructor** that increases module count
89 /// @remark It doesn't get called, unless expicitly called
90 /// @note Doesn't do anything other than increase module count
91 /// </summary>
92 /// <param name="val">The object passed to the copy constructor (we don't do anything with it)</param>
93 __AEModuleBase<T>(const __AEModuleBase<T>& val) noexcept {
94 this->m_ullModulesAlive++;
95 }
96
97 /// <summary>
98 /// **Basic destructor** that decreases module count
99 /// </summary>
100 virtual ~__AEModuleBase<T>(void) noexcept {
101 this->m_ullModulesAlive--;
102 }
103
104 /// <summary>
105 /// **Basic assignment operator -- it does nothing, so it's deleted** (make your own)
106 /// </summary>
107 virtual __AEModuleBase<T>& operator=(const __AEModuleBase<T>& val) = delete;
108
109 /// <summary>
110 /// Get the **amount of the currently-alive instances of module T**
111 /// @see __AEModuleBase<T>::m_ullModulesAlive
112 /// </summary>
113 /// <returns>Unsigned long long of the alive module amount</returns>
114 [[nodiscard]] static inline ullint getModuleAmount(void) noexcept {
115 return m_ullModulesAlive.load();
116 }
117
118 /// <summary>
119 /// Get the **name of the module**
120 /// @attention You *need* to add #REGISTER_MODULE() to the end of the class declarations if you want to use this thing
121 /// @see #REGISTER_MODULE()
122 /// </summary>
123 /// <returns>The name of the module as a const std::strinv_view type</returns>
124 [[nodiscard]] static constexpr const std::string_view getModuleName(void) noexcept;
125
126protected:
127
128 /// <summary>
129 /// Amount of the currently-alive (not destroyed) instances of the module.
130 /// </summary>
131 static inline std::atomic<std::size_t> m_ullModulesAlive = 0;
132
133 //REGISTER_MODULE(__AEModuleBase)
134};
