642inline cint AEFileReader::readIntString(T& num) {
643 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
644 _AEFR_EXIT_ON_READ_CLOSED_FILE;
645
646 std::memset(&num, NULL, sizeof(T));
647 cint temp = 0;
648 if constexpr (IS_SAME_NOC(T, signed char)) {
649 temp = fscanf(m_fpFilestr, "%hhd", &num);
650 }
651 else if constexpr (IS_SAME_NOC(T, unsigned char)) {
652 temp = fscanf(m_fpFilestr, "%hhud", &num);
653 }
654 else if constexpr (IS_SAME_NOC(T, short)) {
655 temp = fscanf(m_fpFilestr, "%hd", &num);
656 }
657 else if constexpr (IS_SAME_NOC(T, unsigned short)) {
658 temp = fscanf(m_fpFilestr, "%hud", &num);
659 }
660 else if constexpr (IS_SAME_NOC(T, int)) {
661 temp = fscanf(m_fpFilestr, "%d", &num);
662 }
663 else if constexpr (IS_SAME_NOC(T, unsigned int)) {
664 temp = fscanf(m_fpFilestr, "%ud", &num);
665 }
666 else if constexpr (IS_SAME_NOC(T, long)) {
667 temp = fscanf(m_fpFilestr, "%ld", &num);
668 }
669 else if constexpr (IS_SAME_NOC(T, unsigned long)) {
670 temp = fscanf(m_fpFilestr, "%lud", &num);
671 }
672 else if constexpr (IS_SAME_NOC(T, long long)) {
673 temp = fscanf(m_fpFilestr, "%lld", &num);
674 }
675 else if constexpr (IS_SAME_NOC(T, std::size_t)) {
676 temp = fscanf(m_fpFilestr, "%zd", &num);
677 }
678 else {
679 temp = fscanf(m_fpFilestr, "%llud", &num);
680 }
681
682 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
683 if (temp != 1) {
684 return AEFR_ERR_READ_EOF;
685 }
686 return AEFR_ERR_READ_SUCCESS;
687}
