596inline cint AEFileReader::readIntString(T& num) {
597 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
598 _AEFR_EXIT_ON_READ_CLOSED_FILE;
599
600 std::memset(&num, NULL, sizeof(T));
601 cint temp = 0;
602 if constexpr (IS_SAME_NOC(T, signed char)) {
603 temp = fscanf(m_fpFilestr, "%hhd", &num);
604 }
605 else if constexpr (IS_SAME_NOC(T, unsigned char)) {
606 temp = fscanf(m_fpFilestr, "%hhud", &num);
607 }
608 else if constexpr (IS_SAME_NOC(T, short)) {
609 temp = fscanf(m_fpFilestr, "%hd", &num);
610 }
611 else if constexpr (IS_SAME_NOC(T, unsigned short)) {
612 temp = fscanf(m_fpFilestr, "%hud", &num);
613 }
614 else if constexpr (IS_SAME_NOC(T, int)) {
615 temp = fscanf(m_fpFilestr, "%d", &num);
616 }
617 else if constexpr (IS_SAME_NOC(T, unsigned int)) {
618 temp = fscanf(m_fpFilestr, "%ud", &num);
619 }
620 else if constexpr (IS_SAME_NOC(T, long)) {
621 temp = fscanf(m_fpFilestr, "%ld", &num);
622 }
623 else if constexpr (IS_SAME_NOC(T, unsigned long)) {
624 temp = fscanf(m_fpFilestr, "%lud", &num);
625 }
626 else if constexpr (IS_SAME_NOC(T, long long)) {
627 temp = fscanf(m_fpFilestr, "%lld", &num);
628 }
629 else if constexpr (IS_SAME_NOC(T, std::size_t)) {
630 temp = fscanf(m_fpFilestr, "%zd", &num);
631 }
632 else {
633 temp = fscanf(m_fpFilestr, "%llud", &num);
634 }
635
636 this->m_ullTotalReads++;
637 if (temp != 1) {
638 this->m_cLastError = AEFR_ERR_READING_EOF;
639 return AEFR_ERR_READING_EOF;
640 }
641 return AEFR_ERR_READ_SUCCESS;
642}
