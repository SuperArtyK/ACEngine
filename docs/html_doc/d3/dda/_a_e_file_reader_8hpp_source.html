609inline cint AEFileReader::readIntString(T& num) {
610 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
611 _AEFR_EXIT_ON_READ_CLOSED_FILE;
612
613 std::memset(&num, NULL, sizeof(T));
614 cint temp = 0;
615 if constexpr (IS_SAME_NOC(T, signed char)) {
616 temp = fscanf(m_fpFilestr, "%hhd", &num);
617 }
618 else if constexpr (IS_SAME_NOC(T, unsigned char)) {
619 temp = fscanf(m_fpFilestr, "%hhud", &num);
620 }
621 else if constexpr (IS_SAME_NOC(T, short)) {
622 temp = fscanf(m_fpFilestr, "%hd", &num);
623 }
624 else if constexpr (IS_SAME_NOC(T, unsigned short)) {
625 temp = fscanf(m_fpFilestr, "%hud", &num);
626 }
627 else if constexpr (IS_SAME_NOC(T, int)) {
628 temp = fscanf(m_fpFilestr, "%d", &num);
629 }
630 else if constexpr (IS_SAME_NOC(T, unsigned int)) {
631 temp = fscanf(m_fpFilestr, "%ud", &num);
632 }
633 else if constexpr (IS_SAME_NOC(T, long)) {
634 temp = fscanf(m_fpFilestr, "%ld", &num);
635 }
636 else if constexpr (IS_SAME_NOC(T, unsigned long)) {
637 temp = fscanf(m_fpFilestr, "%lud", &num);
638 }
639 else if constexpr (IS_SAME_NOC(T, long long)) {
640 temp = fscanf(m_fpFilestr, "%lld", &num);
641 }
642 else if constexpr (IS_SAME_NOC(T, std::size_t)) {
643 temp = fscanf(m_fpFilestr, "%zd", &num);
644 }
645 else {
646 temp = fscanf(m_fpFilestr, "%llud", &num);
647 }
648
649 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
650 if (temp != 1) {
651 return AEFR_ERR_READING_EOF;
652 }
653 return AEFR_ERR_READ_SUCCESS;
654}
