173 /// <summary>
174 /// Reads the given amount of bytes and interprets them as the string.
175 /// @note Modifies the length of the std::string to the dcount+1 size (to accomodate for the trailing null-termiantion character).
176 /// @note If the resulting data size is less than dcount (like from EOF), resizes it to the amount read.
177 /// @note If the file is closed, it doesn't modify the string
178 /// </summary>
179 /// <param name="str">The refernce to the std::string object to read the bytes of string to</param>
180 /// <param name="dcount">The amount of characters to read</param>
181 /// <returns>AEFR_ERR_READ_SUCCESS on success; otherwise AEFR_ERR_FILE_NOT_OPEN if file isn't open, or other AEFR_ERR_READ_* flags (like AEFR_ERR_READ_EOF)</returns>
182 inline cint readString(std::string& str, const std::size_t dcount) {
183 _AEFR_EXIT_ON_READ_CLOSED_FILE;
184 str.resize(dcount + 1);
185 const cint temp = this->readString(str.data(), dcount);
186 if (temp) {
187 str.resize(this->m_szLastReadAmount);
188 }
189 return temp;
190 }

572 /// <summary>
573 /// Sets read cursor position to pos from origin.
574 /// @note If cursor is beyond end of file, next read will trigger EOF error and will not read any data (just fill the given data place with NULL)
575 /// @note If origin is not SEEK_SET, SEEK_CUR or SEEK_END returns AEFR_ERR_READ_EOF
576 /// </summary>

581 _AEFR_EXIT_ON_CLOSED_FILE;
582 if(origin != SEEK_SET && origin != SEEK_CUR && origin != SEEK_END){
583 return AEFR_ERR_READ_EOF;
584 }

643inline cint AEFileReader::readIntString(T& num) noexcept requires(std::is_integral<T>::value == true) {
644 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
645 _AEFR_EXIT_ON_READ_CLOSED_FILE;
646
647 std::memset(&num, AENULL, sizeof(T));
648 cint temp = 0;
649 if constexpr (IS_SAME_NOCV(T, signed char)) {
650 temp = fscanf(m_fpFilestr, "%hhd", &num);
651 }
652 else if constexpr (IS_SAME_NOCV(T, unsigned char)) {
653 temp = fscanf(m_fpFilestr, "%hhud", &num);
654 }
655 else if constexpr (IS_SAME_NOCV(T, short)) {
656 temp = fscanf(m_fpFilestr, "%hd", &num);
657 }
658 else if constexpr (IS_SAME_NOCV(T, unsigned short)) {
659 temp = fscanf(m_fpFilestr, "%hud", &num);
660 }
661 else if constexpr (IS_SAME_NOCV(T, int)) {
662 temp = fscanf(m_fpFilestr, "%d", &num);
663 }
664 else if constexpr (IS_SAME_NOCV(T, unsigned int)) {
665 temp = fscanf(m_fpFilestr, "%ud", &num);
666 }
667 else if constexpr (IS_SAME_NOCV(T, long)) {
668 temp = fscanf(m_fpFilestr, "%ld", &num);
669 }
670 else if constexpr (IS_SAME_NOCV(T, unsigned long)) {
671 temp = fscanf(m_fpFilestr, "%lud", &num);
672 }
673 else if constexpr (IS_SAME_NOCV(T, long long)) {
674 temp = fscanf(m_fpFilestr, "%lld", &num);
675 }
676 else if constexpr (IS_SAME_NOCV(T, std::size_t)) {
677 temp = fscanf(m_fpFilestr, "%zd", &num);
678 }
679 else {
680 temp = fscanf(m_fpFilestr, "%llud", &num);
681 }
682
683 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
684 if (temp != 1) {
685 return AEFR_ERR_READ_EOF;
686 }
687 return AEFR_ERR_READ_SUCCESS;
688}
