558inline cint AEFileReader::readIntString(T& num) {
559 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
560 _AEFR_EXIT_ON_READ_CLOSED_FILE;
561
562 std::memset(&num, NULL, sizeof(T));
563 cint temp = 0;
564 if constexpr (IS_SAME_NOC(T, signed char)) {
565 temp = fscanf(m_fpFilestr, "%hhd", &num);
566 }
567 else if constexpr (IS_SAME_NOC(T, unsigned char)) {
568 temp = fscanf(m_fpFilestr, "%hhud", &num);
569 }
570 else if constexpr (IS_SAME_NOC(T, short)) {
571 temp = fscanf(m_fpFilestr, "%hd", &num);
572 }
573 else if constexpr (IS_SAME_NOC(T, unsigned short)) {
574 temp = fscanf(m_fpFilestr, "%hud", &num);
575 }
576 else if constexpr (IS_SAME_NOC(T, int)) {
577 temp = fscanf(m_fpFilestr, "%d", &num);
578 }
579 else if constexpr (IS_SAME_NOC(T, unsigned int)) {
580 temp = fscanf(m_fpFilestr, "%ud", &num);
581 }
582 else if constexpr (IS_SAME_NOC(T, long)) {
583 temp = fscanf(m_fpFilestr, "%ld", &num);
584 }
585 else if constexpr (IS_SAME_NOC(T, unsigned long)) {
586 temp = fscanf(m_fpFilestr, "%lud", &num);
587 }
588 else if constexpr (IS_SAME_NOC(T, long long)) {
589 temp = fscanf(m_fpFilestr, "%lld", &num);
590 }
591 else if constexpr (IS_SAME_NOC(T, std::size_t)) {
592 temp = fscanf(m_fpFilestr, "%zd", &num);
593 }
594 else {
595 temp = fscanf(m_fpFilestr, "%llud", &num);
596 }
597
598 this->m_ullTotalReads++;
599 if (temp != 1) {
600 this->m_cLastError = AEFR_ERR_READING_EOF;
601 return AEFR_ERR_READING_EOF;
602 }
603 return AEFR_ERR_READ_SUCCESS;
604}
