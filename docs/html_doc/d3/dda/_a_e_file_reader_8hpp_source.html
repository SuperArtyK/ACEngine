177 inline cint readStringNULL(char* str, const std::size_t dcount) {
178 _AEFR_EXIT_ON_CLOSED_FILE;
179
180
181 if (!dcount || !str) {
182 this->m_cLastError = AEFR_ERR_WRITE_ZERO_SIZE;
183 return AEFR_ERR_WRITE_ZERO_SIZE;
184 }
185
186 std::memset(str, NULL, dcount+1);
187 const cint temp = this->readData_ptr(str, 1, dcount);
188
189 char* const nulchar = std::strchr(str, '\0');
190 std::memset(nulchar, NULL, dcount - (nulchar - str));
191
192 fseek(m_fpFilestr, -llint((dcount - (nulchar - str) - 1)), SEEK_CUR);
193
194 return temp;
195 }

291 _AEFR_EXIT_ON_CLOSED_FILE;
292

299 this->m_ullTotalReads++;
