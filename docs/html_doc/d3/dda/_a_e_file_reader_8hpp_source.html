630inline cint AEFileReader::readIntString(T& num) {
631 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
632 _AEFR_EXIT_ON_READ_CLOSED_FILE;
633
634 std::memset(&num, NULL, sizeof(T));
635 cint temp = 0;
636 if constexpr (IS_SAME_NOC(T, signed char)) {
637 temp = fscanf(m_fpFilestr, "%hhd", &num);
638 }
639 else if constexpr (IS_SAME_NOC(T, unsigned char)) {
640 temp = fscanf(m_fpFilestr, "%hhud", &num);
641 }
642 else if constexpr (IS_SAME_NOC(T, short)) {
643 temp = fscanf(m_fpFilestr, "%hd", &num);
644 }
645 else if constexpr (IS_SAME_NOC(T, unsigned short)) {
646 temp = fscanf(m_fpFilestr, "%hud", &num);
647 }
648 else if constexpr (IS_SAME_NOC(T, int)) {
649 temp = fscanf(m_fpFilestr, "%d", &num);
650 }
651 else if constexpr (IS_SAME_NOC(T, unsigned int)) {
652 temp = fscanf(m_fpFilestr, "%ud", &num);
653 }
654 else if constexpr (IS_SAME_NOC(T, long)) {
655 temp = fscanf(m_fpFilestr, "%ld", &num);
656 }
657 else if constexpr (IS_SAME_NOC(T, unsigned long)) {
658 temp = fscanf(m_fpFilestr, "%lud", &num);
659 }
660 else if constexpr (IS_SAME_NOC(T, long long)) {
661 temp = fscanf(m_fpFilestr, "%lld", &num);
662 }
663 else if constexpr (IS_SAME_NOC(T, std::size_t)) {
664 temp = fscanf(m_fpFilestr, "%zd", &num);
665 }
666 else {
667 temp = fscanf(m_fpFilestr, "%llud", &num);
668 }
669
670 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
671 if (temp != 1) {
672 return AEFR_ERR_READ_EOF;
673 }
674 return AEFR_ERR_READ_SUCCESS;
675}
