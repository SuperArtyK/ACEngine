646inline cint AEFileReader::readIntString(T& num) {
647 static_assert(std::is_integral<T>::value, "Cannot use non-integral types in AEFileReader::readIntString()");
648 _AEFR_EXIT_ON_READ_CLOSED_FILE;
649
650 std::memset(&num, NULL, sizeof(T));
651 cint temp = 0;
652 if constexpr (IS_SAME_NOC(T, signed char)) {
653 temp = fscanf(m_fpFilestr, "%hhd", &num);
654 }
655 else if constexpr (IS_SAME_NOC(T, unsigned char)) {
656 temp = fscanf(m_fpFilestr, "%hhud", &num);
657 }
658 else if constexpr (IS_SAME_NOC(T, short)) {
659 temp = fscanf(m_fpFilestr, "%hd", &num);
660 }
661 else if constexpr (IS_SAME_NOC(T, unsigned short)) {
662 temp = fscanf(m_fpFilestr, "%hud", &num);
663 }
664 else if constexpr (IS_SAME_NOC(T, int)) {
665 temp = fscanf(m_fpFilestr, "%d", &num);
666 }
667 else if constexpr (IS_SAME_NOC(T, unsigned int)) {
668 temp = fscanf(m_fpFilestr, "%ud", &num);
669 }
670 else if constexpr (IS_SAME_NOC(T, long)) {
671 temp = fscanf(m_fpFilestr, "%ld", &num);
672 }
673 else if constexpr (IS_SAME_NOC(T, unsigned long)) {
674 temp = fscanf(m_fpFilestr, "%lud", &num);
675 }
676 else if constexpr (IS_SAME_NOC(T, long long)) {
677 temp = fscanf(m_fpFilestr, "%lld", &num);
678 }
679 else if constexpr (IS_SAME_NOC(T, std::size_t)) {
680 temp = fscanf(m_fpFilestr, "%zd", &num);
681 }
682 else {
683 temp = fscanf(m_fpFilestr, "%llud", &num);
684 }
685
686 this->m_ullTotalReads.fetch_add(1, std::memory_order::relaxed);
687 if (temp != 1) {
688 return AEFR_ERR_READ_EOF;
689 }
690 return AEFR_ERR_READ_SUCCESS;
691}
